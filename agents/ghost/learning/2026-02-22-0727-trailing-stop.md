# Trailing Stop Calculator
*Ghost Learning | 2026-02-22*

Dynamic stop adjustment to lock in profits while giving trades room to breathe.

## Types

| Method | Best For | Formula |
|--------|----------|---------|
| **Percentage** | Simple trailing | `high * (1 - pct)` |
| **ATR** | Volatility-aware | `high - (N × ATR)` |
| **Chandelier** | Trend following | `high - (N × ATR)` |
| **Parabolic** | Accelerating exits | `SAR formula` |

## Implementation

```python
"""
Trailing Stop Calculator
Dynamic stop management for profit protection.
"""

from dataclasses import dataclass, field
from decimal import Decimal
from typing import Optional, Literal
from enum import Enum
import math


class StopType(Enum):
    PERCENTAGE = "percentage"
    ATR = "atr"
    CHANDELIER = "chandelier"


@dataclass
class TrailingStop:
    """Trailing stop state tracker."""
    side: Literal["long", "short"]
    method: StopType
    trail_value: Decimal  # % or ATR multiplier
    
    # State
    current_stop: Optional[Decimal] = None
    highest_price: Optional[Decimal] = None
    lowest_price: Optional[Decimal] = None
    activations: int = 0
    
    @property
    def is_active(self) -> bool:
        return self.current_stop is not None
    
    @property
    def distance_pct(self) -> Optional[Decimal]:
        """Current distance from price to stop."""
        if not self.is_active or not self.highest_price:
            return None
        if self.side == "long":
            return (self.highest_price - self.current_stop) / self.highest_price
        else:
            return (self.current_stop - self.lowest_price) / self.lowest_price


class TrailingStopCalculator:
    """
    Calculate and track trailing stops.
    
    Supports:
    - Percentage trailing
    - ATR-based trailing
    - Activation thresholds
    - Step adjustments
    """
    
    def __init__(
        self,
        side: Literal["long", "short"],
        method: StopType = StopType.PERCENTAGE,
        trail_value: Decimal = Decimal("0.05"),  # 5% or 2×ATR
        activation_pct: Optional[Decimal] = None,  # Activate after X% profit
        step_pct: Optional[Decimal] = None,  # Only move stop if price moves X%
        initial_stop: Optional[Decimal] = None,
    ):
        self.side = side
        self.method = method
        self.trail_value = trail_value
        self.activation_pct = activation_pct
        self.step_pct = step_pct
        self.initial_stop = initial_stop
        
        self._state = TrailingStop(
            side=side,
            method=method,
            trail_value=trail_value
        )
        self._entry_price: Optional[Decimal] = None
        self._last_trigger_price: Optional[Decimal] = None
        self._atr: Optional[Decimal] = None
    
    def set_atr(self, atr: Decimal) -> None:
        """Set current ATR value for ATR-based methods."""
        self._atr = atr
    
    def update(
        self,
        high: Decimal,
        low: Decimal,
        close: Decimal
    ) -> dict:
        """
        Update trailing stop with new price data.
        
        Returns dict with stop price and status.
        """
        # Initialize
        if self._entry_price is None:
            self._entry_price = close
            self._state.highest_price = high
            self._state.lowest_price = low
            if self.initial_stop:
                self._state.current_stop = self.initial_stop
            return self._result("initialized")
        
        # Track extremes
        if self.side == "long":
            self._state.highest_price = max(self._state.highest_price or high, high)
        else:
            self._state.lowest_price = min(self._state.lowest_price or low, low)
        
        # Check activation
        if not self._state.is_active:
            if self._should_activate():
                self._activate()
                self._state.activations += 1
                return self._result("activated")
            return self._result("waiting")
        
        # Calculate new stop
        new_stop = self._calculate_stop()
        
        # Apply step filter
        if self.step_pct and self._last_trigger_price:
            price_moved = abs(close - self._last_trigger_price) / self._last_trigger_price
            if price_moved < self.step_pct:
                return self._result("holding", new_stop)
        
        # Update stop (only move in profit direction)
        if self.side == "long":
            if new_stop > self._state.current_stop:
                self._state.current_stop = new_stop
                self._last_trigger_price = close
                return self._result("updated", new_stop)
        else:
            if new_stop < self._state.current_stop:
                self._state.current_stop = new_stop
                self._last_trigger_price = close
                return self._result("updated", new_stop)
        
        return self._result("holding", new_stop)
    
    def _should_activate(self) -> bool:
        """Check if activation threshold met."""
        if not self.activation_pct or not self._entry_price:
            return True  # Immediate activation
        
        if self.side == "long":
            profit_pct = (self._state.highest_price - self._entry_price) / self._entry_price
        else:
            profit_pct = (self._entry_price - self._state.lowest_price) / self._entry_price
        
        return profit_pct >= self.activation_pct
    
    def _activate(self) -> None:
        """Activate trailing stop."""
        self._state.current_stop = self._calculate_stop()
    
    def _calculate_stop(self) -> Decimal:
        """Calculate stop price based on method."""
        if self.method == StopType.PERCENTAGE:
            if self.side == "long":
                return self._state.highest_price * (1 - self.trail_value)
            else:
                return self._state.lowest_price * (1 + self.trail_value)
        
        elif self.method == StopType.ATR:
            if not self._atr:
                raise ValueError("ATR not set for ATR-based trailing")
            if self.side == "long":
                return self._state.highest_price - (self.trail_value * self._atr)
            else:
                return self._state.lowest_price + (self.trail_value * self._atr)
        
        elif self.method == StopType.CHANDELIER:
            if not self._atr:
                raise ValueError("ATR not set for Chandelier exit")
            # Chandelier uses highest high for longs, lowest low for shorts
            if self.side == "long":
                return self._state.highest_price - (self.trail_value * self._atr)
            else:
                return self._state.lowest_price + (self.trail_value * self._atr)
        
        raise ValueError(f"Unknown method: {self.method}")
    
    def _result(self, status: str, new_stop: Optional[Decimal] = None) -> dict:
        """Build result dict."""
        return {
            "status": status,
            "stop_price": float(self._state.current_stop) if self._state.current_stop else None,
            "new_stop": float(new_stop) if new_stop else None,
            "highest": float(self._state.highest_price) if self._state.highest_price else None,
            "lowest": float(self._state.lowest_price) if self._state.lowest_price else None,
            "distance_pct": float(self._state.distance_pct) if self._state.distance_pct else None,
            "activations": self._state.activations,
            "is_active": self._state.is_active
        }
    
    def check_exit(self, close: Decimal) -> bool:
        """Check if price hit stop."""
        if not self._state.is_active:
            return False
        if self.side == "long":
            return close <= self._state.current_stop
        else:
            return close >= self._state.current_stop
    
    @property
    def stop_price(self) -> Optional[Decimal]:
        return self._state.current_stop


# === Quick Functions ===

def pct_trail_stop(
    entry: float,
    high_since_entry: float,
    trail_pct: float = 0.05,
    side: str = "long"
) -> float:
    """One-liner percentage trailing stop."""
    if side == "long":
        return high_since_entry * (1 - trail_pct)
    else:
        return high_since_entry * (1 + trail_pct)


def atr_trail_stop(
    high: float,
    atr: float,
    multiplier: float = 2.0,
    side: str = "long"
) -> float:
    """One-liner ATR trailing stop."""
    if side == "long":
        return high - (multiplier * atr)
    else:
        return high + (multiplier * atr)


# === Usage ===

if __name__ == "__main__":
    print("=== Long Position: 5% Trailing Stop ===")
    print("Activate after 3% profit, 1% step")
    print()
    
    trail = TrailingStopCalculator(
        side="long",
        method=StopType.PERCENTAGE,
        trail_value=Decimal("0.05"),
        activation_pct=Decimal("0.03"),  # Activate at 3% profit
        step_pct=Decimal("0.01"),  # Move only on 1% price moves
        initial_stop=Decimal("95")  # Initial stop below entry
    )
    
    # Simulate price action
    prices = [
        (100, 100, 100),   # Entry
        (102, 101, 101.5), # +1.5%
        (103, 102.5, 103), # +3% - should activate
        (105, 104, 104.5), # +4.5%
        (108, 106, 107),   # +7%
        (107, 105, 105.5), # Pullback
        (106, 104, 104),   # Hit stop?
    ]
    
    print(f"{'High':>8} {'Low':>8} {'Close':>8} {'Status':>12} {'Stop':>10} {'Dist%':>8}")
    print("-" * 55)
    
    for high, low, close in prices:
        result = trail.update(Decimal(str(high)), Decimal(str(low)), Decimal(str(close)))
        hit = trail.check_exit(Decimal(str(close)))
        status = "EXIT!" if hit else result["status"]
        stop = f"${result['stop_price']:.2f}" if result['stop_price'] else "N/A"
        dist = f"{result['distance_pct']*100:.1f}%" if result.get('distance_pct') else "N/A"
        print(f"{high:>8.1f} {low:>8.1f} {close:>8.1f} {status:>12} {stop:>10} {dist:>8}")
    
    print()
    print("=== Quick Reference ===")
    print(f"5% trail on $100: stop at ${pct_trail_stop(100, 105, 0.05):.2f}")
    print(f"2×ATR trail: ATR=$2, High=$110 → stop at ${atr_trail_stop(110, 2, 2.0):.2f}")
```
