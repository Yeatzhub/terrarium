# Kelly Criterion Position Sizer
*Ghost Learning | 2026-02-22*

Calculate optimal position size from edge and odds. With practical safety limits.

## The Formula

```
Kelly % = W - (1-W)/R

W = Win probability (0-1)
R = Win amount / Loss amount (reward/risk ratio)
```

## Implementation

```python
"""
Kelly Criterion Position Sizer
Optimal bet sizing with practical safeguards.
"""

from dataclasses import dataclass
from decimal import Decimal
from typing import Optional
import math


@dataclass
class TradeStats:
    """Historical trade statistics."""
    wins: int
    losses: int
    total_pnl: Decimal
    total_wins: Decimal
    total_losses: Decimal
    
    @property
    def total_trades(self) -> int:
        return self.wins + self.losses
    
    @property
    def win_rate(self) -> Decimal:
        if self.total_trades == 0:
            return Decimal("0")
        return Decimal(self.wins) / Decimal(self.total_trades)
    
    @property
    def avg_win(self) -> Decimal:
        return self.total_wins / self.wins if self.wins > 0 else Decimal("0")
    
    @property
    def avg_loss(self) -> Decimal:
        return self.total_losses / self.losses if self.losses > 0 else Decimal("0")
    
    @property
    def reward_risk(self) -> Decimal:
        if self.avg_loss == 0:
            return Decimal("0")
        return abs(self.avg_win / self.avg_loss)


class KellySizer:
    """
    Kelly Criterion position sizer with practical limits.
    
    Features:
    - Full Kelly, Half Kelly, Quarter Kelly
    - Max position cap
    - Min edge threshold
    - Drawdown-adjusted sizing
    """
    
    def __init__(
        self,
        account_balance: Decimal,
        max_risk_pct: Decimal = Decimal("0.02"),  # Max 2% per trade
        kelly_fraction: Decimal = Decimal("0.50"),  # Half Kelly by default
        min_edge: Decimal = Decimal("0.05"),  # Minimum 5% edge required
    ):
        self.account_balance = account_balance
        self.max_risk_pct = max_risk_pct
        self.kelly_fraction = kelly_fraction
        self.min_edge = min_edge
    
    def kelly_pct(
        self,
        win_prob: Decimal,
        reward_risk: Decimal
    ) -> Decimal:
        """
        Calculate Kelly percentage.
        
        Kelly = W - (1-W)/R
        """
        if reward_risk <= 0:
            return Decimal("0")
        
        kelly = win_prob - (Decimal("1") - win_prob) / reward_risk
        return max(Decimal("0"), kelly)
    
    def adjusted_kelly(
        self,
        win_prob: Decimal,
        reward_risk: Decimal,
        drawdown_pct: Optional[Decimal] = None
    ) -> Decimal:
        """
        Kelly adjusted for fraction and drawdown.
        
        Reduces Kelly during drawdowns.
        """
        kelly = self.kelly_pct(win_prob, reward_risk)
        
        # Apply fraction (half/three-quarter Kelly)
        kelly = kelly * self.kelly_fraction
        
        # Drawdown reduction
        if drawdown_pct and drawdown_pct > 0:
            dd_factor = Decimal("1") - (drawdown_pct * Decimal("1.5"))
            kelly = kelly * max(Decimal("0.25"), dd_factor)
        
        return kelly
    
    def position_size(
        self,
        win_prob: Decimal,
        reward_risk: Decimal,
        entry_price: Decimal,
        stop_price: Decimal,
        drawdown_pct: Optional[Decimal] = None
    ) -> dict:
        """
        Calculate position size.
        
        Returns dict with sizing details.
        """
        # Calculate Kelly
        kelly = self.adjusted_kelly(win_prob, reward_risk, drawdown_pct)
        
        # Check edge threshold
        edge = kelly
        if edge < self.min_edge:
            return {
                "size": Decimal("0"),
                "reason": f"Edge {edge:.2%} below minimum {self.min_edge:.2%}",
                "kelly_pct": kelly,
                "risk_amount": Decimal("0")
            }
        
        # Cap at max risk
        kelly = min(kelly, self.max_risk_pct)
        
        # Calculate risk amount
        risk_amount = self.account_balance * kelly
        
        # Calculate position size from stop distance
        stop_distance = abs(entry_price - stop_price) / entry_price
        if stop_distance == 0:
            return {
                "size": Decimal("0"),
                "reason": "Stop distance is zero",
                "kelly_pct": kelly,
                "risk_amount": Decimal("0")
            }
        
        position_size = risk_amount / stop_distance
        position_value = position_size * entry_price
        
        return {
            "size": position_size,
            "value": position_value,
            "kelly_pct": kelly,
            "raw_kelly": self.kelly_pct(win_prob, reward_risk),
            "risk_amount": risk_amount,
            "risk_pct": kelly,
            "stop_distance_pct": stop_distance,
            "accounts_used_pct": position_value / self.account_balance,
            "reason": "OK"
        }
    
    @classmethod
    def from_stats(
        cls,
        stats: TradeStats,
        account_balance: Decimal,
        **kwargs
    ) -> "KellySizer":
        """Create sizer from historical trade stats."""
        return cls(
            account_balance=account_balance,
            **kwargs
        )
    
    def suggest_size(
        self,
        stats: TradeStats,
        entry_price: Decimal,
        stop_price: Decimal,
        drawdown_pct: Optional[Decimal] = None
    ) -> dict:
        """Suggest position size based on historical stats."""
        return self.position_size(
            win_prob=stats.win_rate,
            reward_risk=stats.reward_risk,
            entry_price=entry_price,
            stop_price=stop_price,
            drawdown_pct=drawdown_pct
        )


def quick_kelly(win_rate: float, avg_win: float, avg_loss: float) -> float:
    """Quick Kelly calculation helper."""
    if avg_loss == 0:
        return 0.0
    rr = abs(avg_win / avg_loss)
    kelly = win_rate - (1 - win_rate) / rr
    return max(0.0, kelly)


# === Usage ===

if __name__ == "__main__":

    # Example 1: From known probabilities
    sizer = KellySizer(
        account_balance=Decimal("100000"),
        max_risk_pct=Decimal("0.02"),
        kelly_fraction=Decimal("0.50"),  # Half Kelly

    )
    
    result = sizer.position_size(
        win_prob=Decimal("0.60"),      # 60% win rate
        reward_risk=Decimal("1.5"),    # Win 1.5x what you risk
        entry_price=Decimal("50000"),
        stop_price=Decimal("49000"),   # 2% stop
    )
    
    print("=== Kelly Sizing ===")
    print(f"Raw Kelly:    {result['raw_kelly']:.2%}")
    print(f"Adjusted:     {result['kelly_pct']:.2%}")
    print(f"Risk Amount:  ${result['risk_amount']:,.2f}")
    print(f"Position:     {result['size']:.4f} units")
    print(f"Position Val: ${result['value']:,.2f}")
    print(f"Account %:    {result['accounts_used_pct']:.2%}")
    
    # Example 2: From trade history
    stats = TradeStats(
        wins=45,
        losses=30,
        total_pnl=Decimal("15000"),
        total_wins=Decimal("22500"),
        total_losses=Decimal("-7500")
    )
    
    print(f"\n=== From Stats ===")
    print(f"Win Rate:     {stats.win_rate:.2%}")
    print(f"Avg Win:      ${stats.avg_win:,.2f}")
    print(f"Avg Loss:     ${stats.avg_loss:,.2f}")
    print(f"Reward/Risk:  {stats.reward_risk:.2f}")
    print(f"Quick Kelly:  {quick_kelly(float(stats.win_rate), float(stats.avg_win), float(stats.avg_loss)):.2%}")
    
    # Quick reference table
    print("\n=== Kelly Reference Table (Half Kelly) ===")
    print("Win% | R:R=1.0 | R:R=1.5 | R:R=2.0 | R:R=3.0")
    print("-" * 45)
    for win_pct in [0.50, 0.55, 0.60, 0.65, 0.70]:
        row = f"{win_pct:.0%}  |"
        for rr in [1.0, 1.5, 2.0, 3.0]:
            k = quick_kelly(win_pct, rr, 1.0) * 0.5
            row += f" {k:>6.1%}  |"
        print(row)
```

## Quick Rules

| Win Rate | Min R:R Needed | Notes |
|----------|---------------|-------|
| 50% | 1.0 | Break-even edge |
| 55% | 0.8 | Low edge |
| 60% | 0.67 | Solid edge |
| 65% | 0.54 | Strong edge |

## Safety Features

1. **Kelly Fraction** - Use 0.25-0.50 Kelly (quarter/half) for safety
2. **Max Risk Cap** - Never exceed max_risk_pct (default 2%)
3. **Min Edge** - Skip trades below minimum edge
4. **Drawdown Adjust** - Reduce size during drawdowns
