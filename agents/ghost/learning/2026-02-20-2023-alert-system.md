# Alert System — Trading Signal Notifications

**Purpose:** Monitor price/indicator conditions and trigger alerts when thresholds are crossed  
**Use Case:** Get notified of setups, stops hit, targets reached, or unusual activity

## The Code

```python
"""
Alert System
Configurable alert engine for price levels, indicators, and custom conditions.
Supports multiple notification channels with rate limiting and cooldowns.
"""

from dataclasses import dataclass, field
from typing import List, Dict, Optional, Callable, Any, Set
from datetime import datetime, timedelta
from enum import Enum, auto
import time


class AlertType(Enum):
    PRICE_ABOVE = auto()
    PRICE_BELOW = auto()
    PRICE_RANGE = auto()
    PERCENT_CHANGE = auto()
    VOLUME_SPIKE = auto()
    RSI_OVERBOUGHT = auto()
    RSI_OVERSOLD = auto()
    MA_CROSSOVER = auto()
    MA_CROSSUNDER = auto()
    CUSTOM = auto()


class AlertPriority(Enum):
    LOW = auto()
    MEDIUM = auto()
    HIGH = auto()
    CRITICAL = auto()


class AlertStatus(Enum):
    ACTIVE = auto()
    TRIGGERED = auto()
    ACKNOWLEDGED = auto()
    EXPIRED = auto()
    DISABLED = auto()


@dataclass
class Alert:
    """Single alert definition."""
    id: str
    name: str
    symbol: str
    alert_type: AlertType
    priority: AlertPriority
    
    # Trigger conditions
    condition_params: Dict[str, Any]
    
    # Timing
    created_at: datetime
    expires_at: Optional[datetime] = None
    cooldown_minutes: int = 60
    
    # State
    status: AlertStatus = AlertStatus.ACTIVE
    last_triggered: Optional[datetime] = None
    trigger_count: int = 0
    
    # Actions
    message_template: str = ""
    actions: List[Callable] = field(default_factory=list)
    
    def is_on_cooldown(self) -> bool:
        if not self.last_triggered:
            return False
        return datetime.now() < self.last_triggered + timedelta(minutes=self.cooldown_minutes)
    
    def can_trigger(self) -> bool:
        if self.status != AlertStatus.ACTIVE:
            return False
        if self.expires_at and datetime.now() > self.expires_at:
            return False
        return not self.is_on_cooldown()


@dataclass
class AlertEvent:
    """Triggered alert event."""
    alert_id: str
    alert_name: str
    symbol: str
    alert_type: AlertType
    priority: AlertPriority
    
    triggered_at: datetime
    trigger_value: float
    trigger_message: str
    
    # Context
    price: float
    volume: Optional[float] = None
    additional_data: Dict = field(default_factory=dict)


@dataclass
class PriceData:
    """Current market data for alert checking."""
    symbol: str
    timestamp: datetime
    price: float
    volume: float
    
    # Optional indicators
    rsi: Optional[float] = None
    sma_20: Optional[float] = None
    sma_50: Optional[float] = None
    change_pct: Optional[float] = None
    
    # Additional context
    high_24h: Optional[float] = None
    low_24h: Optional[float] = None
    avg_volume: Optional[float] = None


class NotificationChannel:
    """Base class for notification channels."""
    
    def __init__(self, name: str, min_priority: AlertPriority = AlertPriority.LOW):
        self.name = name
        self.min_priority = min_priority
        self.rate_limit_minutes = 1
        self.last_sent: Dict[str, datetime] = {}
    
    def can_send(self, alert: Alert) -> bool:
        if alert.priority.value < self.min_priority.value:
            return False
        
        last = self.last_sent.get(alert.symbol)
        if last and datetime.now() < last + timedelta(minutes=self.rate_limit_minutes):
            return False
        
        return True
    
    def send(self, event: AlertEvent) -> bool:
        """Send notification. Override in subclasses."""
        raise NotImplementedError
    
    def _mark_sent(self, symbol: str):
        self.last_sent[symbol] = datetime.now()


class ConsoleChannel(NotificationChannel):
    """Print alerts to console."""
    
    def send(self, event: AlertEvent) -> bool:
        emoji = {
            AlertPriority.LOW: "⚪",
            AlertPriority.MEDIUM: "🔵",
            AlertPriority.HIGH: "🟠",
            AlertPriority.CRITICAL: "🔴"
        }.get(event.priority, "⚪")
        
        print(f"\n{emoji} ALERT: {event.alert_name}")
        print(f"   Symbol: {event.symbol}")
        print(f"   Type: {event.alert_type.name}")
        print(f"   Price: ${event.price:.2f}")
        print(f"   Trigger: {event.trigger_message}")
        print(f"   Time: {event.triggered_at.strftime('%H:%M:%S')}")
        
        self._mark_sent(event.symbol)
        return True


class AlertSystem:
    """
    Multi-channel alert system for trading signals.
    
    Usage:
        alerts = AlertSystem()
        alerts.add_channel(ConsoleChannel("console"))
        
        # Create alerts
        alerts.add_alert(Alert(
            id="aapl_breakout",
            name="AAPL Above 180",
            symbol="AAPL",
            alert_type=AlertType.PRICE_ABOVE,
            priority=AlertPriority.HIGH,
            condition_params={"price": 180.0}
        ))
        
        # Check price updates
        alerts.check_price(PriceData("AAPL", datetime.now(), 181.50, 1000000))
    """
    
    def __init__(self):
        self.alerts: Dict[str, Alert] = {}
        self.channels: List[NotificationChannel] = []
        self.event_history: List[AlertEvent] = []
        self.max_history = 1000
    
    def add_channel(self, channel: NotificationChannel):
        """Add notification channel."""
        self.channels.append(channel)
    
    def add_alert(self, alert: Alert) -> str:
        """Add alert to monitor."""
        self.alerts[alert.id] = alert
        return alert.id
    
    def remove_alert(self, alert_id: str) -> bool:
        """Remove alert."""
        if alert_id in self.alerts:
            del self.alerts[alert_id]
            return True
        return False
    
    def check_price(self, price_data: PriceData) -> List[AlertEvent]:
        """Check all alerts against price data."""
        triggered = []
        
        symbol_alerts = [a for a in self.alerts.values() if a.symbol == price_data.symbol]
        
        for alert in symbol_alerts:
            if not alert.can_trigger():
                continue
            
            event = self._evaluate_alert(alert, price_data)
            if event:
                triggered.append(event)
                self._trigger_alert(alert, event)
        
        return triggered
    
    def check_batch(self, price_data_list: List[PriceData]) -> List[AlertEvent]:
        """Check multiple price updates."""
        all_triggered = []
        for data in price_data_list:
            all_triggered.extend(self.check_price(data))
        return all_triggered
    
    def _evaluate_alert(self, alert: Alert, data: PriceData) -> Optional[AlertEvent]:
        """Evaluate single alert against price data."""
        params = alert.condition_params
        
        if alert.alert_type == AlertType.PRICE_ABOVE:
            threshold = params.get("price")
            if threshold and data.price >= threshold:
                return AlertEvent(
                    alert_id=alert.id,
                    alert_name=alert.name,
                    symbol=data.symbol,
                    alert_type=alert.alert_type,
                    priority=alert.priority,
                    triggered_at=datetime.now(),
                    trigger_value=data.price,
                    trigger_message=f"Price above ${threshold}",
                    price=data.price,
                    volume=data.volume
                )
        
        elif alert.alert_type == AlertType.PRICE_BELOW:
            threshold = params.get("price")
            if threshold and data.price <= threshold:
                return AlertEvent(
                    alert_id=alert.id,
                    alert_name=alert.name,
                    symbol=data.symbol,
                    alert_type=alert.alert_type,
                    priority=alert.priority,
                    triggered_at=datetime.now(),
                    trigger_value=data.price,
                    trigger_message=f"Price below ${threshold}",
                    price=data.price,
                    volume=data.volume
                )
        
        elif alert.alert_type == AlertType.RSI_OVERSOLD:
            threshold = params.get("rsi", 30)
            if data.rsi and data.rsi <= threshold:
                return AlertEvent(
                    alert_id=alert.id,
                    alert_name=alert.name,
                    symbol=data.symbol,
                    alert_type=alert.alert_type,
                    priority=alert.priority,
                    triggered_at=datetime.now(),
                    trigger_value=data.rsi,
                    trigger_message=f"RSI oversold at {data.rsi:.1f}",
                    price=data.price,
                    volume=data.volume
                )
        
        elif alert.alert_type == AlertType.RSI_OVERBOUGHT:
            threshold = params.get("rsi", 70)
            if data.rsi and data.rsi >= threshold:
                return AlertEvent(
                    alert_id=alert.id,
                    alert_name=alert.name,
                    symbol=data.symbol,
                    alert_type=alert.alert_type,
                    priority=alert.priority,
                    triggered_at=datetime.now(),
                    trigger_value=data.rsi,
                    trigger_message=f"RSI overbought at {data.rsi:.1f}",
                    price=data.price,
                    volume=data.volume
                )
        
        elif alert.alert_type == AlertType.PERCENT_CHANGE:
            threshold = params.get("change_pct")
            if threshold and data.change_pct:
                if abs(data.change_pct) >= abs(threshold):
                    direction = "up" if data.change_pct > 0 else "down"
                    return AlertEvent(
                        alert_id=alert.id,
                        alert_name=alert.name,
                        symbol=data.symbol,
                        alert_type=alert.alert_type,
                        priority=alert.priority,
                        triggered_at=datetime.now(),
                        trigger_value=data.change_pct,
                        trigger_message=f"Price moved {data.change_pct:+.1f}%",
                        price=data.price,
                        volume=data.volume
                    )
        
        elif alert.alert_type == AlertType.VOLUME_SPIKE:
            threshold = params.get("multiplier", 2.0)
            if data.avg_volume and data.volume >= data.avg_volume * threshold:
                ratio = data.volume / data.avg_volume
                return AlertEvent(
                    alert_id=alert.id,
                    alert_name=alert.name,
                    symbol=data.symbol,
                    alert_type=alert.alert_type,
                    priority=alert.priority,
                    triggered_at=datetime.now(),
                    trigger_value=ratio,
                    trigger_message=f"Volume spike {ratio:.1f}x average",
                    price=data.price,
                    volume=data.volume
                )
        
        elif alert.alert_type == AlertType.MA_CROSSOVER:
            if data.sma_20 and data.sma_50:
                prev_20 = params.get("prev_sma_20", data.sma_20 * 0.99)
                prev_50 = params.get("prev_sma_50", data.sma_50 * 1.01)
                
                was_below = prev_20 < prev_50
                now_above = data.sma_20 > data.sma_50
                
                if was_below and now_above:
                    params["prev_sma_20"] = data.sma_20
                    params["prev_sma_50"] = data.sma_50
                    return AlertEvent(
                        alert_id=alert.id,
                        alert_name=alert.name,
                        symbol=data.symbol,
                        alert_type=alert.alert_type,
                        priority=alert.priority,
                        triggered_at=datetime.now(),
                        trigger_value=data.price,
                        trigger_message="SMA-20 crossed above SMA-50",
                        price=data.price,
                        volume=data.volume
                    )
        
        return None
    
    def _trigger_alert(self, alert: Alert, event: AlertEvent):
        """Process triggered alert."""
        alert.last_triggered = datetime.now()
        alert.trigger_count += 1
        alert.status = AlertStatus.TRIGGERED
        
        self.event_history.append(event)
        
        # Keep history bounded
        if len(self.event_history) > self.max_history:
            self.event_history = self.event_history[-self.max_history:]
        
        # Send to channels
        for channel in self.channels:
            if channel.can_send(alert):
                try:
                    channel.send(event)
                except Exception:
                    pass
        
        # Run custom actions
        for action in alert.actions:
            try:
                action(event)
            except Exception:
                pass
    
    def acknowledge(self, alert_id: str) -> bool:
        """Acknowledge triggered alert."""
        if alert_id in self.alerts:
            self.alerts[alert_id].status = AlertStatus.ACKNOWLEDGED
            return True
        return False
    
    def reset(self, alert_id: str) -> bool:
        """Reset alert to active."""
        if alert_id in self.alerts:
            self.alerts[alert_id].status = AlertStatus.ACTIVE
            return True
        return False
    
    def disable(self, alert_id: str) -> bool:
        """Disable alert."""
        if alert_id in self.alerts:
            self.alerts[alert_id].status = AlertStatus.DISABLED
            return True
        return False
    
    def get_active_alerts(self) -> List[Alert]:
        """Get all active alerts."""
        return [a for a in self.alerts.values() if a.status == AlertStatus.ACTIVE]
    
    def get_triggered_alerts(self) -> List[Alert]:
        """Get recently triggered alerts."""
        return [a for a in self.alerts.values() if a.status == AlertStatus.TRIGGERED]
    
    def get_stats(self) -> Dict:
        """Get alert system statistics."""
        total = len(self.alerts)
        active = sum(1 for a in self.alerts.values() if a.status == AlertStatus.ACTIVE)
        triggered = sum(1 for a in self.alerts.values() if a.status == AlertStatus.TRIGGERED)
        total_triggers = sum(a.trigger_count for a in self.alerts.values())
        
        return {
            "total_alerts": total,
            "active": active,
            "triggered": triggered,
            "total_triggers": total_triggers,
            "recent_events": len(self.event_history)
        }

def example_basic_alerts():
    """Demonstrate basic alert functionality."""
    print("=" * 70)
    print("Alert System Demo - Basic Price Alerts")
    print("=" * 70)
    
    alerts = AlertSystem()
    alerts.add_channel(ConsoleChannel("console", min_priority=AlertPriority.LOW))
    
    # Create price alerts
    alerts.add_alert(Alert(
        id="aapl_breakout",
        name="AAPL Breakout Alert",
        symbol="AAPL",
        alert_type=AlertType.PRICE_ABOVE,
        priority=AlertPriority.HIGH,
        condition_params={"price": 180.0},
        cooldown_minutes=30
    ))
    
    alerts.add_alert(Alert(
        id="tsla_stop",
        name="TSLA Stop Loss",
        symbol="TSLA",
        alert_type=AlertType.PRICE_BELOW,
        priority=AlertPriority.CRITICAL,
        condition_params={"price": 240.0}
    ))
    
    alerts.add_alert(Alert(
        id="spy_vol",
        name="SPY Volume Spike",
        symbol="SPY",
        alert_type=AlertType.VOLUME_SPIKE,
        priority=AlertPriority.MEDIUM,
        condition_params={"multiplier": 2.0}
    ))
    
    print(f"\nAdded {len(alerts.alerts)} alerts")
    
    # Simulate price updates
    scenarios = [
        ("AAPL", 179.50, 1000000),
        ("AAPL", 180.25, 1200000),  # Should trigger breakout
        ("TSLA", 245.00, 800000),
        ("TSLA", 239.50, 900000),  # Should trigger stop
        ("SPY", 450.00, 35000000),  # High volume - should trigger
        ("AAPL", 180.50, 1100000),  # Cooldown - should NOT trigger
    ]
    
    print("\n--- Simulating Price Updates ---")
    for symbol, price, volume in scenarios:
        data = PriceData(symbol, datetime.now(), price, volume)
        triggered = alerts.check_price(data)
        
        if not triggered:
            print(f"{symbol} @ ${price:.2f} - No alert")
        
        time.sleep(0.1)
    
    # Stats
    print("\n" + "=" * 70)
    print("Alert Statistics:")
    print("=" * 70)
    stats = alerts.get_stats()
    for key, value in stats.items():
        print(f"  {key}: {value}")
```

## Alert Types

| Type | Trigger | Use Case |
|------|---------|----------|
| **PRICE_ABOVE** | Price ≥ level | Breakout entry |
| **PRICE_BELOW** | Price ≤ level | Stop loss, support break |
| **PRICE_RANGE** | Price in range | Range-bound detection |
| **PERCENT_CHANGE** | % move threshold | Unusual activity |
| **VOLUME_SPIKE** | Volume N× avg | Confirm moves |
| **RSI_OVERSOLD** | RSI ≤ 30 | Mean reversion buy |
| **RSI_OVERBOUGHT** | RSI ≥ 70 | Mean reversion sell |
| **MA_CROSSOVER** | SMA cross up | Trend start |
| **MA_CROSSUNDER** | SMA cross down | Trend end |
| **CUSTOM** | User-defined | Any condition |

## Priority Levels

| Priority | Use |
|----------|-----|
| **CRITICAL** | Stop loss hit, major level |
| **HIGH** | Entry signals, targets |
| **MEDIUM** | Volume spikes, patterns |
| **LOW** | Info, daily summaries |

## Cooldown & Rate Limits

- **Alert cooldown:** Prevents same alert triggering repeatedly (spike protection)
- **Channel rate limit:** Limits notifications per symbol per minute
- **Expiry:** Alerts auto-expire after set time

## Why This Matters

- **Don't watch charts 24/7** — Alerts notify you only when something happens
- **Protect positions** — Get notified when stops are hit immediately
- **Capture setups** — Breakout alerts find entries you might miss
- **Avoid noise** — Cooldowns prevent alert spam during choppy action
- **Confirm moves** — Volume + price alerts confirm legitimate breaks
