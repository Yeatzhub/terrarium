"""
Trade Report Generator v2
Performance analysis with enhanced metrics.

import csv
import json
from dataclasses import dataclass, field
from datetime import datetime
from decimal import Decimal
from pathlib import Path
from typing import Optional, List
from statistics import mean, median, stdev


@dataclass
class Trade:
    """Single trade record."""
    symbol: str
    side: str
    entry: Decimal
    exit: Decimal
    size: Decimal
    pnl: Decimal
    timestamp: Optional[datetime] = None
    r_multiple: Optional[Decimal] = None
    
    @classmethod
    def from_csv(cls, row: dict) -> "Trade":
        """Parse from CSV row."""
        ts = row.get("timestamp")
        return cls(
            symbol=row.get("symbol", "").upper(),
            side=row.get("side", "").lower(),
            entry=Decimal(str(row.get("entry", row.get("entry_price", 0)))),
            exit=Decimal(str(row.get("exit", row.get("exit_price", 0)))),
            size=Decimal(str(row.get("size", 0))),
            pnl=Decimal(str(row.get("pnl", row.get("profit", 0)))),
            timestamp=datetime.fromisoformat(ts) if ts else None,
            r_multiple=Decimal(str(row["r_multiple"])) if "r_multiple" in row else None
        )
    
    @property
    def is_winner(self) -> bool:
        return self.pnl > 0


@dataclass
class PerformanceMetrics:
    """Aggregated performance metrics."""
    total_trades: int = 0
    winning_trades: int = 0
    losing_trades: int = 0
    breakeven_trades: int = 0
    
    win_rate: float = 0.0
    gross_profit: Decimal = Decimal("0")
    gross_loss: Decimal = Decimal("0")
    net_pnl: Decimal = Decimal("0")
    
    avg_winner: Decimal = Decimal("0")
    avg_loser: Decimal = Decimal("0")
    avg_trade: Decimal = Decimal("0")
    median_trade: Decimal = Decimal("0")
    
    profit_factor: float = 0.0
    payoff_ratio: float = 0.0  # avg win / avg loss
    
    best_trade: Decimal = Decimal("0")
    worst_trade: Decimal = Decimal("0")
    
    max_consecutive_wins: int = 0
    max_consecutive_losses: int = 0
    current_streak: int = 0  # Positive = wins, negative = losses
    
    std_dev: float = 0.0
    sharpe_ratio: float = 0.0
    sortino_ratio: float = 0.0
    
    # Enhanced metrics
    max_drawdown: Decimal = Decimal("0")
    max_drawdown_pct: float = 0.0
    calmar_ratio: float = 0.0
    avg_r_multiple: Decimal = Decimal("0")


@dataclass
class SymbolStats:
    """Per-symbol performance."""
    symbol: str
    trades: int
    winners: int
    losers: int
    win_rate: float
    pnl: Decimal
    avg_trade: Decimal


class TradeReportGenerator:
    """Generate comprehensive trade reports."""
    
    def __init__(self, trades: List[Trade]):
        self.trades = trades
        self.metrics = self._calculate_metrics()
        self.by_symbol = self._by_symbol_stats()
    
    def _calculate_metrics(self) -> PerformanceMetrics:
        """Calculate all performance metrics."""
        m = PerformanceMetrics()
        
        if not self.trades:
            return m
        
        # Basic counts
        m.total_trades = len(self.trades)
        winners = [t for t in self.trades if t.pnl > 0]
        losers = [t for t in self.trades if t.pnl < 0]
        breakeven = [t for t in self.trades if t.pnl == 0]
        
        m.winning_trades = len(winners)
        m.losing_trades = len(losers)
        m.breakeven_trades = len(breakeven)
        m.win_rate = m.winning_trades / m.total_trades * 100
        
        # P&L
        m.gross_profit = sum(t.pnl for t in winners) or Decimal("0")
        m.gross_loss = sum(abs(t.pnl) for t in losers) or Decimal("0")
        m.net_pnl = sum(t.pnl for t in self.trades)
        
        # Averages
        m.avg_winner = m.gross_profit / m.winning_trades if m.winning_trades else Decimal("0")
        m.avg_loser = m.gross_loss / m.losing_trades if m.losing_trades else Decimal("0")
        m.avg_trade = m.net_pnl / m.total_trades
        
        pnls = [float(t.pnl) for t in self.trades]
        m.median_trade = Decimal(str(median(pnls)))
        
        # Profit factor
        m.profit_factor = float(m.gross_profit / m.gross_loss) if m.gross_loss > 0 else float('inf')
        m.payoff_ratio = float(m.avg_winner / m.avg_loser) if m.avg_loser > 0 else 0.0
        
        # Best/worst
        m.best_trade = max(t.pnl for t in self.trades)
        m.worst_trade = min(t.pnl for t in self.trades)
        
        # Streaks - FIXED
        m.max_consecutive_wins, m.max_consecutive_losses, m.current_streak = self._calculate_streaks()
        
        # Risk metrics
        if len(pnls) >= 2:
            m.std_dev = stdev(pnls)
            
            # Sharpe (annualized, assuming daily trades)
            avg_return = mean(pnls)
            m.sharpe_ratio = (avg_return / m.std_dev * (252 ** 0.5)) if m.std_dev > 0 else 0.0
            
            # Sortino (downside deviation only)
            neg_returns = [r for r in pnls if r < 0]
            if neg_returns:
                downside_std = (sum(r**2 for r in neg_returns) / len(neg_returns)) ** 0.5
                m.sortino_ratio = (avg_return / downside_std * (252 ** 0.5)) if downside_std > 0 else 0.0
        
        # Drawdown
        m.max_drawdown, m.max_drawdown_pct = self._calculate_drawdown()
        
        # Calmar
        annual_return = float(m.net_pnl) * 252 / m.total_trades if m.total_trades > 0 else 0
        m.calmar_ratio = annual_return / m.max_drawdown_pct if m.max_drawdown_pct > 0 else 0.0
        
        # R-multiples
        r_values = [t.r_multiple for t in self.trades if t.r_multiple is not None]
        m.avg_r_multiple = sum(r_values) / len(r_values) if r_values else Decimal("0")
        
        return m
    
    def _calculate_streaks(self) -> tuple[int, int, int]:
        """Calculate consecutive win/loss streaks.
        
        FIXED: Properly track streaks without resetting incorrectly.
        """
        if not self.trades:
            return 0, 0, 0
        
        max_wins = 0
        max_losses = 0
        current_wins = 0
        current_losses = 0
        
        for t in self.trades:
            if t.pnl > 0:  # Win
                current_wins += 1
                current_losses = 0
                max_wins = max(max_wins, current_wins)
            elif t.pnl < 0:  # Loss
                current_losses += 1
                current_wins = 0
                max_losses = max(max_losses, current_losses)
            # Breakeven: doesn't break streak in this version
        
        # Current streak (positive = wins, negative = losses)
        current = current_wins if current_wins > 0 else -current_losses
        
        return max_wins, max_losses, current
    
    def _calculate_drawdown(self) -> tuple[Decimal, float]:
        """Calculate maximum drawdown from equity curve."""
        if not self.trades:
            return Decimal("0"), 0.0
        
        peak = Decimal("0")
        max_dd = Decimal("0")
        equity = Decimal("0")
        
        for t in self.trades:
            equity += t.pnl
            if equity > peak:
                peak = equity
            dd = peak - equity
            if dd > max_dd:
                max_dd = dd
        
        dd_pct = float(max_dd / peak * 100) if peak > 0 else 0.0
        return max_dd, dd_pct
    
    def _by_symbol_stats(self) -> List[SymbolStats]:
        """Breakdown by trading symbol."""
        from collections import defaultdict
        
        symbol_trades = defaultdict(list)
        for t in self.trades:
            symbol_trades[t.symbol].append(t)
        
        stats = []
        for symbol, trades in sorted(symbol_trades.items()):
            winners = [t for t in trades if t.pnl > 0]
            pnl = sum(t.pnl for t in trades)
            stats.append(SymbolStats(
                symbol=symbol,
                trades=len(trades),
                winners=len(winners),
                losers=len(trades) - len(winners),
                win_rate=len(winners) / len(trades) * 100,
                pnl=pnl,
                avg_trade=pnl / len(trades)
            ))
        
        return stats
    
    def generate_markdown(self) -> str:
        """Generate comprehensive markdown report."""
        m = self.metrics
        
        report = f"""# Trading Performance Report
*Generated: {datetime.now().strftime("%Y-%m-%d %H:%M")}*

## 📊 Summary

| Metric | Value |
|--------|-------|
| Total Trades | {m.total_trades} |
| Winners / Losers / BE | {m.winning_trades} / {m.losing_trades} / {m.breakeven_trades} |
| Win Rate | {m.win_rate:.1f}% |
| **Net P&L** | ${m.net_pnl:+,.2f} |
| Profit Factor | {m.profit_factor:.2f} |
| Payoff Ratio | {m.payoff_ratio:.2f} |

## 💰 Trade Statistics

| Metric | Winners | Losers |
|--------|---------|--------|
| Count | {m.winning_trades} | {m.losing_trades} |
| Total | ${m.gross_profit:,.2f} | ${m.gross_loss:,.2f} |
| Average | ${m.avg_winner:,.2f} | ${m.avg_loser:,.2f} |
| Best / Worst | ${m.best_trade:+,.2f} | ${m.worst_trade:,.2f} |

## 📈 Performance Metrics

| Metric | Value |
|--------|-------|
| Average Trade | ${m.avg_trade:+,.2f} |
| Median Trade | ${m.median_trade:+,.2f} |
| Std Dev | ${m.std_dev:,.2f} |
| Sharpe Ratio | {m.sharpe_ratio:.2f} |
| Sortino Ratio | {m.sortino_ratio:.2f} |
| Max Drawdown | ${m.max_drawdown:,.2f} ({m.max_drawdown_pct:.1f}%) |
| Calmar Ratio | {m.calmar_ratio:.2f} |
"""
        
        # Add R-multiples if available
        if m.avg_r_multiple != 0:
            report += f"| Avg R-Multiple | {m.avg_r_multiple:.2f}R |\n"
        
        report += f"""
## 🔥 Streaks

| Metric | Value |
|--------|-------|
| Max Consecutive Wins | {m.max_consecutive_wins} |
| Max Consecutive Losses | {m.max_consecutive_losses} |
| Current Streak | {"+" if m.current_streak > 0 else ""}{m.current_streak} |
"""
        
        # By-symbol breakdown
        if self.by_symbol:
            report += "\n## 📋 By Symbol\n\n"
            report += "| Symbol | Trades | W/L | Win% | P&L | Avg |\n"
            report += "|--------|--------|-----|------|-----|-----|\n"
            for s in self.by_symbol:
                report += f"| {s.symbol} | {s.trades} | {s.winners}/{s.losers} | {s.win_rate:.0f}% | ${s.pnl:+,.2f} | ${s.avg_trade:+,.2f} |\n"
        
        # Recent trades
        report += "\n## 📝 Recent Trades\n\n"
        report += "| Symbol | Side | Entry | Exit | Size | P&L |\n"
        report += "|--------|------|-------|------|------|-----|\n"
        for t in self.trades[-10:]:
            pnl_str = f"${t.pnl:+,.2f}"
            report += f"| {t.symbol} | {t.side} | ${t.entry:,.2f} | ${t.exit:,.2f} | {t.size} | {pnl_str} |\n"
        
        return report
    
    def summary_line(self) -> str:
        """One-line summary for notifications."""
        m = self.metrics
        return (
            f"{m.total_trades} trades | {m.win_rate:.0f}% win | "
            f"${m.net_pnl:+,.0f} | PF {m.profit_factor:.1f} | "
            f"Sharpe {m.sharpe_ratio:.1f}"
        )


def load_trades_from_csv(path: Path) -> List[Trade]:
    """Load trades from CSV file."""
    trades = []
    with open(path, "r") as f:
        reader = csv.DictReader(f)
        for row in reader:
            trades.append(Trade.from_csv(row))
    return trades
