"""
Rate Limiter Pattern
Token bucket and sliding window implementations for API rate limiting.
"""

import asyncio
import time
from dataclasses import dataclass, field
from typing import Optional, Callable, Awaitable
from collections import deque


@dataclass
class RateLimitConfig:
    """Rate limit configuration."""
    requests_per_second: float = 10.0
    burst_size: Optional[int] = None    # Max burst allowed
    wait_timeout: float = 30.0          # Max wait time for slot


class TokenBucket:
    """
    Token bucket rate limiter.
    Allows bursts up to bucket size, then steady rate.
    """
    
    def __init__(
        self,
        rate: float,           # Tokens per second
        capacity: Optional[int] = None  # Bucket size (burst)
    ):
        self.rate = rate
        self.capacity = capacity or int(rate * 2)  # Default 2s of burst
        self.tokens = self.capacity
        self.last_update = time.monotonic()
        self._lock = asyncio.Lock()
    
    async def acquire(self, tokens: int = 1) -> bool:
        """Acquire tokens. Returns True if available."""
        async with self._lock:
            now = time.monotonic()
            elapsed = now - self.last_update
            
            # Refill tokens
            self.tokens = min(
                self.capacity,
                self.tokens + elapsed * self.rate
            )
            self.last_update = now
            
            if self.tokens >= tokens:
                self.tokens -= tokens
                return True
            return False
    
    async def wait_for_token(self, timeout: float = 30.0) -> None:
        """Wait until a token is available."""
        start = time.monotonic()
        while True:
            if await self.acquire():
                return
            
            if time.monotonic() - start > timeout:
                raise TimeoutError(f"Rate limit wait exceeded {timeout}s")
            
            # Calculate wait time
            wait = (1 - self.tokens) / self.rate
            await asyncio.sleep(min(wait, 0.1))


class SlidingWindow:
    """
    Sliding window rate limiter.
    Tracks requests in a time window.
    """
    
    def __init__(self, max_requests: int, window_seconds: float):
        self.max_requests = max_requests
        self.window = window_seconds
        self.requests: deque[float] = deque()
        self._lock = asyncio.Lock()
    
    async def acquire(self) -> bool:
        """Check if request is allowed."""
        async with self._lock:
            now = time.monotonic()
            cutoff = now - self.window
            
            # Remove old requests
            while self.requests and self.requests[0] < cutoff:
                self.requests.popleft()
            
            if len(self.requests) < self.max_requests:
                self.requests.append(now)
                return True
            return False
    
    async def wait_for_slot(self, timeout: float = 30.0) -> None:
        """Wait until a slot is available."""
        start = time.monotonic()
        while True:
            if await self.acquire():
                return
            
            if time.monotonic() - start > timeout:
                raise TimeoutError(f"Rate limit wait exceeded {timeout}s")
            
            # Wait for oldest request to expire
            if self.requests:
                wait = self.requests[0] + self.window - time.monotonic()
                await asyncio.sleep(max(0.01, wait))
            else:
                await asyncio.sleep(0.1)


class MultiTierRateLimiter:
    """
    Multi-tier rate limiter for APIs with multiple limits.
    Example: 10 req/s, 100 req/min, 1000 req/hour
    """
    
    def __init__(self):
        self.limiters: list[tuple[str, TokenBucket | SlidingWindow]] = []
    
    def add_limit(
        self,
        name: str,
        max_requests: int,
        window_seconds: float,
        use_token_bucket: bool = False
    ) -> "MultiTierRateLimiter":
        """Add a rate limit tier."""
        if use_token_bucket:
            rate = max_requests / window_seconds
            limiter = TokenBucket(rate=rate, capacity=max_requests)
        else:
            limiter = SlidingWindow(max_requests, window_seconds)
        
        self.limiters.append((name, limiter))
        return self
    
    async def acquire(self, timeout: float = 30.0) -> dict[str, bool]:
        """Try to acquire from all limiters."""
        results = {}
        
        # Check all limits
        for name, limiter in self.limiters:
            if isinstance(limiter, TokenBucket):
                results[name] = await limiter.acquire()
            else:
                results[name] = await limiter.acquire()
        
        return results
    
    async def wait_all(self, timeout: float = 30.0) -> None:
        """Wait until all limiters allow."""
        start = time.monotonic()
        
        while True:
            results = await self.acquire()
            
            if all(results.values()):
                return
            
            if time.monotonic() - start > timeout:
                raise TimeoutError(f"Multi-tier rate limit wait exceeded {timeout}s")
            
            await asyncio.sleep(0.05)


def rate_limit(
    limiter: TokenBucket | SlidingWindow,
    on_limit: Optional[Callable[[], Awaitable[None]]] = None
):
    """Decorator for rate limiting async functions."""
    def decorator(func: Callable[..., Awaitable]) -> Callable[..., Awaitable]:
        async def wrapper(*args, **kwargs):
            if hasattr(limiter, 'wait_for_token'):
                await limiter.wait_for_token()
            else:
                await limiter.wait_for_slot()
            
            return await func(*args, **kwargs)
        return wrapper
    return decorator


class RateLimitedClient:
    """
    Example rate-limited API client.
    Shows common exchange limits.
    """
    
    def __init__(self):
        # Common exchange limits

async def example_token_bucket():
    """Token bucket allows bursts then steady rate."""
    bucket = TokenBucket(rate=5, capacity=10)  # 5/s, burst of 10
    
    # Burst of 10 immediate requests
    for i in range(10):
        acquired = await bucket.acquire()
        print(f"Request {i+1}: {'✓' if acquired else '✗'}")
    
    # Next requests wait
    print("\nAfter burst, rate limited to 5/s:")
    for i in range(3):
        start = time.time()
        await bucket.wait_for_token()
        print(f"Request {i+1}: waited {time.time()-start:.2f}s")


async def example_sliding_window():
    """Sliding window enforces exact limit."""
    window = SlidingWindow(max_requests=3, window_seconds=1.0)
    
    print("Max 3 requests per second:")
    for i in range(5):
        acquired = await window.acquire()
        print(f"Request {i+1}: {'✓' if acquired else '✗'}")
    
    print("\nWaiting 1 second...")
    await asyncio.sleep(1.0)
    
    print("After wait:")
    for i in range(3):
        acquired = await window.acquire()
        print(f"Request {i+1}: {'✓' if acquired else '✗'}")


async def example_decorator():

## Rate Limiter Types

| Type | Best For | Behavior |
|------|----------|----------|
| **Token Bucket** | APIs with burst support | Allows burst, then steady rate |
| **Sliding Window** | Strict limit APIs | Exact count in window |
| **Multi-Tier** | Exchanges (Binance, etc.) | Multiple limits per request |

## Key Differences

- **Token Bucket**: Good for bursty traffic, smoother over time
- **Sliding Window**: Exact enforcement, no bursts beyond limit
- **Multi-Tier**: Use when API has multiple overlapping limits
