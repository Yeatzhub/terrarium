from dataclasses import dataclass, field

from datetime import datetime, timedelta

class TradeEmotion(Enum):
    CONFIDENT = auto()
    FEARFUL = auto()
    GREEDY = auto()
    IMPATIENT = auto()
    HESITANT = auto()
    REVENGEFUL = auto()
    CALM = auto()
    BORED = auto()


class SetupType(Enum):
    BREAKOUT = auto()
    PULLBACK = auto()
    TREND_FOLLOWING = auto()
    MEAN_REVERSION = auto()
    NEWS_BASED = auto()
    EARNINGS = auto()
    TECHNICAL = auto()
    FUNDAMENTAL = auto()


class ExitReason(Enum):
    TARGET_HIT = auto()
    STOP_LOSS = auto()
    TRAILING_STOP = auto()
    TIME_STOP = auto()
    MANUAL = auto()
    REVERSAL_SIGNAL = auto()
    RISK_OFF = auto()
    FORCED = auto()  # Margin call, etc


@dataclass
class TradeEntry:
    """Complete trade record with context."""
    id: str
    symbol: str
    entry_date: datetime
    exit_date: Optional[datetime] = None
    
    # Trade details
    side: str = "long"  # long, short
    entry_price: float = 0.0
    exit_price: Optional[float] = None
    quantity: float = 0.0
    position_size_pct: float = 0.0  # % of account
    
    # Setup
    setup_type: Optional[SetupType] = None
    timeframe: str = "day"  # scalp, day, swing, position
    strategy_name: str = ""
    entry_trigger: str = ""
    
    # Context
    market_condition: str = ""  # trending, ranging, volatile
    sector_trend: str = ""  # strong, weak, neutral
    overall_market: str = ""  # bull, bear, chop
    
    # Emotional state
    pre_trade_emotion: Optional[TradeEmotion] = None
    post_trade_emotion: Optional[TradeEmotion] = None
    confidence_level: int = 5  # 1-10
    
    # Exit
    exit_reason: Optional[ExitReason] = None
    planned_target: Optional[float] = None
    planned_stop: Optional[float] = None
    
    # Outcome
    realized_pnl: float = 0.0
    realized_pnl_pct: float = 0.0  # % of account
    max_favorable_excursion: float = 0.0  # Best price reached
    max_adverse_excursion: float = 0.0  # Worst price reached
    
    # Review
    tags: Set[str] = field(default_factory=set)
    notes: str = ""
    mistakes: List[str] = field(default_factory=list)
    lessons: List[str] = field(default_factory=list)
    would_take_again: Optional[bool] = None
    
    # Timing
    planned_duration: Optional[timedelta] = None
    actual_duration: Optional[timedelta] = None
    
    @property
    def r_multiple(self) -> float:
        """Calculate R-multiple of trade."""
        if self.planned_stop and self.planned_stop != self.entry_price:
            initial_risk = abs(self.entry_price - self.planned_stop)
            if initial_risk > 0:
                return self.realized_pnl / (initial_risk * self.quantity) if self.quantity > 0 else 0
        return 0
    
    @property
    is_winner(self) -> bool:
        return self.realized_pnl > 0
    
    @property
    is_breakeven(self) -> bool:
        return abs(self.realized_pnl) < 0.01
    
    @property
    target_distance_pct(self) -> float:
        """How close did we get to target?"""
        if self.planned_target and self.max_favorable_excursion > 0:
            total_distance = abs(self.planned_target - self.entry_price)
            reached = abs(self.max_favorable_excursion - self.entry_price)
            return (reached / total_distance * 100) if total_distance > 0 else 0
        return 0
    
    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "symbol": self.symbol,
            "date": self.entry_date.isoformat(),
            "side": self.side,
            "pnl": self.realized_pnl,
            "pnl_pct": self.realized_pnl_pct,
            "setup": self.setup_type.name if self.setup_type else None,
            "emotion": self.pre_trade_emotion.name if self.pre_trade_emotion else None,
            "exit": self.exit_reason.name if self.exit_reason else None,
            "r": self.r_multiple,
            "tags": list(self.tags),
            "notes": self.notes
        }


@dataclass
class PatternAnalysis:
    """Discovered patterns in trading behavior."""
    pattern_name: str
    win_rate: float
    avg_return: float
    total_trades: int
    confidence: float  # Statistical confidence
    insight: str


@dataclass
class JournalStats:
    """Comprehensive journal statistics."""
    total_trades: int
    win_rate: float
    avg_pnl: float
    total_pnl: float
    
    # By category
    by_setup: Dict[str, Dict]
    by_emotion: Dict[str, Dict]
    by_exit: Dict[str, Dict]
    by_timeframe: Dict[str, Dict]
    
    # Behavioral
    avg_r_multiple: float
    avg_confidence_when_right: float
    avg_confidence_when_wrong: float
    
    # Mistakes
    common_mistakes: List[Tuple[str, int, float]]  # mistake, count, cost
    
    # Patterns
    discovered_patterns: List[PatternAnalysis]
    
    # Edge
    recent_20_win_rate: float  # Last 20 trades
    recent_50_win_rate: float  # Last 50 trades
    win_rate_trend: str  # improving, declining, stable


class TradeJournal:
    """
    Comprehensive trade journaling system.
    
    Usage:
        journal = TradeJournal()
        
        entry = journal.create_entry("AAPL", 150.0, 100, "long")
        entry.setup_type = SetupType.BREAKOUT
        entry.pre_trade_emotion = TradeEmotion.CONFIDENT
        entry.tags = {"breakout", "tech", "earnings"}
        
        journal.close_entry(entry.id, 155.0, ExitReason.TARGET_HIT)
        
        stats = journal.analyze()
        print(f"Win rate when confident: {stats.by_emotion['CONFIDENT']['win_rate']:.1%}")
    """
    
    def __init__(self, trader_name: str = "trader"):
        self.trader_name = trader_name
        self.entries: Dict[str, TradeEntry] = {}
        self.tags_index: Dict[str, Set[str]] = defaultdict(set)
        self.setup_index: Dict[str, Set[str]] = defaultdict(set)
        self._entry_counter = 0
    
    def create_entry(
        self,
        symbol: str,
        entry_price: float,
        quantity: float,
        side: str = "long",
        entry_date: Optional[datetime] = None
    ) -> TradeEntry:
        """Create new trade entry."""
        self._entry_counter += 1
        entry_id = f"{symbol}_{self._entry_counter}_{datetime.now().strftime('%Y%m%d')}"
        
        entry = TradeEntry(
            id=entry_id,
            symbol=symbol,
            entry_date=entry_date or datetime.now(),
            side=side,
            entry_price=entry_price,
            quantity=quantity
        )
        
        self.entries[entry_id] = entry
        return entry
    
    def close_entry(
        self,
        entry_id: str,
        exit_price: float,
        exit_reason: ExitReason,
        exit_date: Optional[datetime] = None,
        **kwargs
    ) -> TradeEntry:
        """Close trade entry with outcome."""
        if entry_id not in self.entries:
            raise ValueError(f"Entry {entry_id} not found")
        
        entry = self.entries[entry_id]
        entry.exit_date = exit_date or datetime.now()
        entry.exit_price = exit_price
        entry.exit_reason = exit_reason
        
        # Calculate PnL
        if entry.side == "long":
            entry.realized_pnl = (exit_price - entry.entry_price) * entry.quantity
        else:
            entry.realized_pnl = (entry.entry_price - exit_price) * entry.quantity
        
        # Update from kwargs
        for key, value in kwargs.items():
            if hasattr(entry, key):
                setattr(entry, key, value)
        
        # Index
        if entry.setup_type:
            self.setup_index[entry.setup_type.name].add(entry_id)
        for tag in entry.tags:
            self.tags_index[tag].add(entry_id)
        
        return entry
    
    def add_review(self, entry_id: str, notes: str, mistakes: List[str] = None, lessons: List[str] = None):
        """Add post-trade review."""
        if entry_id not in self.entries:
            return
        
        entry = self.entries[entry_id]
        entry.notes = notes
        if mistakes:
            entry.mistakes = mistakes
        if lessons:
            entry.lessons = lessons
    
    def get_entries(
        self,
        symbol: Optional[str] = None,
        setup: Optional[SetupType] = None,
        emotion: Optional[TradeEmotion] = None,
        tags: Optional[Set[str]] = None,
        winner_only: Optional[bool] = None,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None
    ) -> List[TradeEntry]:
        """Filter entries by criteria."""
        results = list(self.entries.values())
        
        if symbol:
            results = [e for e in results if e.symbol == symbol]
        
        if setup:
            results = [e for e in results if e.setup_type == setup]
        
        if emotion:
            results = [e for e in results if e.pre_trade_emotion == emotion]
        
        if tags:
            results = [e for e in results if tags.intersection(e.tags)]
        
        if winner_only is not None:
            results = [e for e in results if e.is_winner == winner_only]
        
        if start_date:
            results = [e for e in results if e.entry_date >= start_date]
        
        if end_date:
            results = [e for e in results if e.entry_date <= end_date]
        
        return sorted(results, key=lambda e: e.entry_date)
    
    def analyze(self) -> JournalStats:
        """Comprehensive analysis of all trades."""
        closed = [e for e in self.entries.values() if e.exit_date]
        
        if not closed:
            return JournalStats(0, 0, 0, 0, {}, {}, {}, {}, 0, 0, 0, [], [], 0, 0, "unknown")
        
        # Overall stats
        winners = [e for e in closed if e.is_winner]
        total_pnl = sum(e.realized_pnl for e in closed)
        
        # By setup
        by_setup = self._analyze_by_category(closed, lambda e: e.setup_type.name if e.setup_type else "unknown")
        
        # By emotion
        by_emotion = self._analyze_by_category(closed, lambda e: e.pre_trade_emotion.name if e.pre_trade_emotion else "unknown")
        
        # By exit reason
        by_exit = self._analyze_by_category(closed, lambda e: e.exit_reason.name if e.exit_reason else "unknown")
        
        # By timeframe
        by_timeframe = self._analyze_by_category(closed, lambda e: e.timeframe)
        
        # R-multiples
        r_multiples = [e.r_multiple for e in closed if e.r_multiple != 0]
        avg_r = statistics.mean(r_multiples) if r_multiples else 0
        
        # Confidence correlation
        right_confidence = [e.confidence_level for e in winners if e.confidence_level]
        wrong_confidence = [e.confidence_level for e in closed if not e.is_winner and e.confidence_level]
        
        # Common mistakes
        all_mistakes = []
        for e in closed:
            if e.mistakes:
                cost = abs(e.realized_pnl) if not e.is_winner else 0
                all_mistakes.extend([(m, cost) for m in e.mistakes])
        
        mistake_counts = Counter(m for m, _ in all_mistakes)
        mistake_costs = defaultdict(float)
        for m, c in all_mistakes:
            mistake_costs[m] += c
        
        common_mistakes = [
            (m, mistake_counts[m], mistake_costs[m])
            for m in mistake_counts.keys()
        ]
        common_mistakes.sort(key=lambda x: x[2], reverse=True)  # Sort by cost
        
        # Recent performance
        recent_20 = closed[-20:] if len(closed) >= 20 else closed
        recent_50 = closed[-50:] if len(closed) >= 50 else closed
        
        recent_20_wr = len([e for e in recent_20 if e.is_winner]) / len(recent_20) if recent_20 else 0
        recent_50_wr = len([e for e in recent_50 if e.is_winner]) / len(recent_50) if recent_50 else 0
        
        # Trend
        if len(closed) >= 50:
            older_50 = closed[-100:-50] if len(closed) >= 100 else closed[:len(closed)//2]
            older_wr = len([e for e in older_50 if e.is_winner]) / len(older_50) if older_50 else 0
            
            if recent_50_wr > older_wr + 0.1:
                trend = "improving"
            elif recent_50_wr < older_wr - 0.1:
                trend = "declining"
            else:
                trend = "stable"
        else:
            trend = "insufficient_data"
        
        # Pattern discovery
        patterns = self._discover_patterns(closed)
        
        return JournalStats(
            total_trades=len(closed),
            win_rate=len(winners) / len(closed),
            avg_pnl=total_pnl / len(closed),
            total_pnl=total_pnl,
            by_setup=by_setup,
            by_emotion=by_emotion,
            by_exit=by_exit,
            by_timeframe=by_timeframe,
            avg_r_multiple=avg_r,
            avg_confidence_when_right=statistics.mean(right_confidence) if right_confidence else 0,
            avg_confidence_when_wrong=statistics.mean(wrong_confidence) if wrong_confidence else 0,
            common_mistakes=common_mistakes[:5],
            discovered_patterns=patterns,
            recent_20_win_rate=recent_20_wr,
            recent_50_win_rate=recent_50_wr,
            win_rate_trend=trend
        )
    
    def _analyze_by_category(
        self,
        entries: List[TradeEntry],
        key_fn: Callable[[TradeEntry], str]
    ) -> Dict[str, Dict]:
        """Analyze trades grouped by category."""
        grouped = defaultdict(list)
        for e in entries:
            grouped[key_fn(e)].append(e)
        
        result = {}
        for key, trades in grouped.items():
            winners = [e for e in trades if e.is_winner]
            pnls = [e.realized_pnl for e in trades]
            
            result[key] = {
                "count": len(trades),
                "win_rate": len(winners) / len(trades) if trades else 0,
                "total_pnl": sum(pnls),
                "avg_pnl": statistics.mean(pnls) if pnls else 0,
                "avg_r": statistics.mean([e.r_multiple for e in trades if e.r_multiple != 0]) or 0
            }
        
        return result
    
    def _discover_patterns(self, entries: List[TradeEntry]) -> List[PatternAnalysis]:
        """Discover patterns in trading data."""
        patterns = []
        
        # Pattern: High confidence trades
        high_conf = [e for e in entries if e.confidence_level >= 8]
        low_conf = [e for e in entries if e.confidence_level <= 4]
        
        if high_conf and low_conf:
            high_wr = len([e for e in high_conf if e.is_winner]) / len(high_conf)
            low_wr = len([e for e in low_conf if e.is_winner]) / len(low_conf)
            
            if high_wr > low_wr + 0.2:
                patterns.append(PatternAnalysis(
                    pattern_name="High confidence edge",
                    win_rate=high_wr,
                    avg_return=statistics.mean([e.realized_pnl for e in high_conf]),
                    total_trades=len(high_conf),
                    confidence=0.8,
                    insight="Trades with 8+ confidence have significantly higher win rate"
                ))
        
        # Pattern: Revenge trading
        if len(entries) >= 10:
            for i in range(1, len(entries)):
                prev, curr = entries[i-1], entries[i]
                if not prev.is_winner and not curr.is_winner:
                    time_diff = (curr.entry_date - prev.exit_date).total_seconds() / 60 if prev.exit_date else 0
                    if 0 < time_diff < 30:  # Within 30 minutes
                        patterns.append(PatternAnalysis(
                            pattern_name="Revenge trading detected",
                            win_rate=0.0,
                            avg_return=sum([prev.realized_pnl, curr.realized_pnl]),
                            total_trades=2,
                            confidence=0.6,
                            insight="Rapid re-entry after loss often leads to second loss"
                        ))
                        break
        
        # Pattern: Setup performance
        for setup_name, setup_stats in self._analyze_by_category(entries, lambda e: e.setup_type.name if e.setup_type else "unknown").items():
            if setup_stats["count"] >= 5 and setup_stats["win_rate"] > 0.6:
                patterns.append(PatternAnalysis(
                    pattern_name=f"Strong {setup_name} setup",
                    win_rate=setup_stats["win_rate"],
                    avg_return=setup_stats["avg_pnl"],
                    total_trades=setup_stats["count"],
                    confidence=min(0.9, setup_stats["count"] / 20),
                    insight=f"{setup_name} setups show consistent edge"
                ))
        
        return patterns
    
    def generate_report(self, days: int = 30) -> str:
        """Generate text report of recent performance."""
        cutoff = datetime.now() - timedelta(days=days)
        recent = self.get_entries(start_date=cutoff)
        stats = self.analyze()
        
        lines = [
            f"{'=' * 60}",
            f"TRADE JOURNAL REPORT - Last {days} days",
            f"{'=' * 60}",
            "",
            f"Trades: {len(recent)} | Win Rate: {stats.win_rate:.1%} | PnL: ${stats.total_pnl:+,.2f}",
            f"Avg R: {stats.avg_r_multiple:.2f} | Trend: {stats.win_rate_trend.upper()}",
            "",
            "Performance by Setup:",
        ]
        
        for setup, data in sorted(stats.by_setup.items(), key=lambda x: x[1]["total_pnl"], reverse=True):
            emoji = "✅" if data["win_rate"] > 0.5 else "❌"
            lines.append(f"  {emoji} {setup:<15} WR: {data['win_rate']:.1%} PnL: ${data['total_pnl']:+,.0f}")
        
        lines.extend(["", "Emotional Impact:"])
        for emotion, data in stats.by_emotion.items():
            if data["count"] >= 3:
                emoji = "✅" if data["win_rate"] > 0.5 else "⚠️" if data["win_rate"] > 0.4 else "❌"
                lines.append(f"  {emoji} {emotion:<12} WR: {data['win_rate']:.1%} ({data['count']} trades)")
        
        if stats.common_mistakes:
            lines.extend(["", "Top Mistakes (by cost):"])
            for mistake, count, cost in stats.common_mistakes[:3]:
                lines.append(f"  • {mistake}: {count}x (${cost:,.0f})")
        
        if stats.discovered_patterns:
            lines.extend(["", "Discovered Patterns:"])
            for p in stats.discovered_patterns[:3]:
                lines.append(f"  💡 {p.pattern_name}: {p.insight}")
        
        lines.append(f"\n{'=' * 60}")
        
        return "\n".join(lines)
    
    def export_json(self, filepath: str):
        """Export all entries to JSON."""
        data = [e.to_dict() for e in self.entries.values()]
        with open(filepath, 'w') as f:
            json.dump(data, f, indent=2)

def example_journal():
    """Demonstrate trade journaling."""
    print("=" * 70)
    print("Trade Journal Demo")
    print("=" * 70)
    
    journal = TradeJournal("Trader_John")
    
    # Simulate a week of trading
    base_date = datetime(2024, 1, 15)
    
    trades_data = [
        # Winners
        ("AAPL", 150, 155, 100, "long", SetupType.BREAKOUT, TradeEmotion.CONFIDENT, ExitReason.TARGET_HIT, 8, 500),
        ("MSFT", 300, 310, 50, "long", SetupType.TREND_FOLLOWING, TradeEmotion.CALM, ExitReason.TARGET_HIT, 7, 500),
        ("NVDA", 400, 420, 40, "long", SetupType.BREAKOUT, TradeEmotion.CONFIDENT, ExitReason.TRAILING_STOP, 9, 800),
        
        # Losers
        ("TSLA", 250, 245, 60, "long", SetupType.MEAN_REVERSION, TradeEmotion.GREEDY, ExitReason.STOP_LOSS, 6, -300),
        ("META", 300, 290, 50, "long", SetupType.NEWS_BASED, TradeEmotion.FEARFUL, ExitReason.STOP_LOSS, 4, -500),
        ("AMD", 100, 95, 100, "long", SetupType.TECHNICAL, TradeEmotion.HESITANT, ExitReason.MANUAL, 5, -500),
        ("NFLX", 500, 485, 30, "long", SetupType.BREAKOUT, TradeEmotion.REVENGEFUL, ExitReason.STOP_LOSS, 7, -450),
    ]
    
    print("\nRecording trades...\n")
    
    for i, (sym, entry, exit, qty, side, setup, emotion, exit_r, conf, pnl) in enumerate(trades_data):
        date = base_date + timedelta(days=i)
        
        entry_obj = journal.create_entry(sym, entry, qty, side, date)
        entry_obj.setup_type = setup
        entry_obj.pre_trade_emotion = emotion
        entry_obj.confidence_level = conf
        entry_obj.planned_target = entry * 1.05
        entry_obj.planned_stop = entry * 0.97
        entry_obj.tags = {setup.name.lower(), sym.lower(), "tech" if sym in ["AAPL", "MSFT", "NVDA"] else "other"}
        
        # Add context notes
        if emotion == TradeEmotion.REVENGEFUL:
            entry_obj.notes = "Took this trade too soon after previous loss"
            entry_obj.mistakes = ["revenge_trading", "poor_timing"]
        elif emotion == TradeEmotion.GREEDY:
            entry_obj.notes = "Chased the move, entry was too late"
            entry_obj.mistakes = ["chasing", "fomo"]
        elif emotion == TradeEmotion.FEARFUL:
            entry_obj.notes = "Sized down due to fear, still stopped out"
            entry_obj.mistakes = ["emotional_sizing"]
        
        journal.close_entry(
            entry_obj.id, exit, exit_r,
            date + timedelta(hours=2),
            realized_pnl=pnl,
            max_favorable_excursion=exit + 2 if pnl > 0 else exit - 2,
            max_adverse_excursion=entry * 0.98 if pnl > 0 else exit
        )
        
        emoji = "✅" if pnl > 0 else "❌"
        print(f"{emoji} {sym}: ${pnl:+,.0f} | {setup.name} | {emotion.name} | Conf: {conf}/10")
    
    # Analysis
    print("\n" + "=" * 70)
    print("ANALYSIS")
    print("=" * 70)
    
    stats = journal.analyze()
    
    print(f"\nOverall Performance:")
    print(f"  Trades: {stats.total_trades}")
    print(f"  Win Rate: {stats.win_rate:.1%}")
    print(f"  Total PnL: ${stats.total_pnl:+,.0f}")
    print(f"  Avg R-Multiple: {stats.avg_r_multiple:.2f}")
    print(f"  Trend: {stats.win_rate_trend.upper()}")
    
    print(f"\nBy Setup Type:")
    for setup, data in sorted(stats.by_setup.items(), key=lambda x: x[1]["win_rate"], reverse=True):
        if data["count"] > 0:
            print(f"  {setup:<15} {data['win_rate']:.1%} ({data['count']} trades, ${data['total_pnl']:+,.0f})")
    
    print(f"\nBy Emotion:")
    for emotion, data in stats.by_emotion.items():
        if data["count"] > 0:
            emoji = "✅" if data["win_rate"] > 0.5 else "⚠️" if data["win_rate"] > 0.4 else "❌"
            print(f"  {emoji} {emotion:<12} {data['win_rate']:.1%} ({data['count']} trades)")
    
    print(f"\nConfidence Analysis:")
    print(f"  Avg confidence on wins: {stats.avg_confidence_when_right:.1f}/10")
    print(f"  Avg confidence on losses: {stats.avg_confidence_when_wrong:.1f}/10")
    
    if stats.common_mistakes:
        print(f"\nTop Mistakes:")
        for mistake, count, cost in stats.common_mistakes:
            print(f"  • {mistake}: {count} times (${cost:,.0f})")
    
    if stats.discovered_patterns:
        print(f"\nDiscovered Patterns:")
        for p in stats.discovered_patterns:
            print(f"  💡 {p.pattern_name}")
            print(f"      {p.insight}")
            print(f"      Win rate: {p.win_rate:.1%} over {p.total_trades} trades")
    
    # Generate report
    print("\n" + "=" * 70)
    print(journal.generate_report(days=30))


def example_edge_detection():
    """Show how journal detects edge erosion."""
    print("\n" + "=" * 70)
    print("Edge Erosion Detection Demo")
    print("=" * 70)
    
    journal = TradeJournal("Erosion_Test")
    base_date = datetime(2024, 1, 1)
    
    # First 30 trades: Strong edge (60% win rate)
    print("\nPhase 1: Strong performance (60% win rate)")
    for i in range(30):
        date = base_date + timedelta(days=i)
        is_win = i % 5 != 0  # 24 wins, 6 losses = 80% but varying size
        pnl = 200 if is_win else -150
        if i % 3 == 0:
            pnl = 300  # Some bigger winners
        
        e = journal.create_entry("SPY", 400, 100, "long", date)
        e.setup_type = SetupType.TREND_FOLLOWING
        e.pre_trade_emotion = TradeEmotion.CALM
        e.confidence_level = 7
        journal.close_entry(e.id, 405 if is_win else 398, ExitReason.TARGET_HIT if is_win else ExitReason.STOP_LOSS, date + timedelta(hours=1), realized_pnl=pnl)
    
    # Next 20 trades: Edge erodes (30% win rate)
    print("\nPhase 2: Edge erosion (30% win rate)")
    for i in range(30, 50):
        date = base_date + timedelta(days=i)
        is_win = i % 3 == 0  # 30% win rate
        pnl = 200 if is_win else -250  # Bigger losses
        
        e = journal.create_entry("SPY", 400, 100, "long", date)
        e.setup_type = SetupType.TREND_FOLLOWING
        e.pre_trade_emotion = TradeEmotion.BORED if i > 40 else TradeEmotion.CALM  # Boredom creeps in
        e.confidence_level = 6 if i > 40 else 7
        journal.close_entry(e.id, 405 if is_win else 395, ExitReason.TARGET_HIT if is_win else ExitReason.STOP_LOSS, date + timedelta(hours=1), realized_pnl=pnl)
    
    stats = journal.analyze()
    
    print(f"\nOverall: {stats.win_rate:.1%} win rate")
    print(f"Recent 20 trades: {stats.recent_20_win_rate:.1%} win rate")
    print(f"Recent 50 trades: {stats.recent_50_win_rate:.1%} win rate")

## Review Questions

After every trade, ask:
1. **Did I follow my plan?** (Entry, stop, target)
2. **What was my emotional state?** (Confident, fearful, greedy)
3. **Was the setup valid?** (High-quality or forced)
4. **What did I learn?** (One lesson per trade)
5. **Would I take this again?** (Yes/No/Maybe)

## Weekly Review Process

1. **Win rate by setup** — Double down on what works
2. **Emotional audit** — Identify destructive patterns
3. **Mistake tally** — Fix one mistake per week
4. **Edge check** — Recent 20 vs overall win rate
5. **Journal review** — Read your notes, spot patterns
