## Pattern 1: Robust Async Task Runner with Timeout

```python
import asyncio
from typing import Any, Callable

async def run_with_timeout(
    coro: Callable[..., Any], 
    timeout: float = 30.0,
    fallback: Any = None
) -> Any:
    """Run a coroutine with timeout and fallback."""
    try:
        return await asyncio.wait_for(coro, timeout=timeout)
    except asyncio.TimeoutError:
        print(f"⚠️ Task timed out after {timeout}s")
        return fallback
    except Exception as e:
        print(f"❌ Task failed: {e}")
        return fallback

# Usage
result = await run_with_timeout(fetch_data(), timeout=10.0, fallback=[])
```

## Pattern 2: Async Context Manager for Resource Cleanup

```python
from contextlib import asynccontextmanager
import aiohttp

@asynccontextmanager
async def managed_session(timeout: int = 30):
    """HTTP session that auto-cleans up, even on failures."""
    session = aiohttp.ClientSession(
        timeout=aiohttp.ClientTimeout(total=timeout)
    )
    try:
        yield session
    finally:
        await session.close()

# Usage
async with managed_session(timeout=10) as session:
    async with session.get("https://api.example.com") as resp:
        data = await resp.json()
```

## Pattern 3: Batch Async Operations with Semaphore

```python
async def batch_process(items: list, max_concurrent: int = 5):
    """Process items with controlled concurrency."""
    semaphore = asyncio.Semaphore(max_concurrent)
    
    async def process_one(item):
        async with semaphore:
            await asyncio.sleep(0.1)  # Simulate work
            return item * 2
    
    tasks = [process_one(item) for item in items]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Filter out exceptions
    return [r for r in results if not isinstance(r, Exception)]
```

## Pattern 4: Graceful Shutdown Handler

```python
import signal

class AsyncWorker:
    def __init__(self):
        self._tasks: set[asyncio.Task] = set()
        self._shutdown = asyncio.Event()
    
    async def run(self):
        loop = asyncio.get_event_loop()
        for sig in (signal.SIGTERM, signal.SIGINT):
            loop.add_signal_handler(sig, self._signal_handler)
        
        try:
            await self._shutdown.wait()
        finally:
            await self._cleanup()
    
    def _signal_handler(self):
        print("🛑 Shutdown signal received")
        self._shutdown.set()
    
    async def _cleanup(self):
        print(f"🧹 Cleaning up {len(self._tasks)} tasks...")
        for task in self._tasks:
            task.cancel()
        await asyncio.gather(*self._tasks, return_exceptions=True)
```

## Quick Reference: Exception Hierarchy

```
Exception
├── asyncio.CancelledError      # Task was cancelled
├── asyncio.TimeoutError        # wait_for() exceeded
├── ConnectionError
│   ├── BrokenPipeError
│   └── ConnectionResetError
└── OSError                      # Network/system errors
```

**Rule of thumb:** Always use `return_exceptions=True` with `gather()` for batch ops.

## TL;DR — Three Rules

1. **Always timebox** → Use `wait_for()` or `timeout=` params
2. **Always cleanup** → Use `@asynccontextmanager` for resources
3. **Always handle cancellation** → Wrap blocking ops, check `asyncio.current_task().cancelled()`
