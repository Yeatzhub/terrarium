# Retry Pattern with Exponential Backoff

**Purpose:** Handle transient failures gracefully with automatic retries and exponential backoff  
**Use Case:** API calls, database connections, network requests—any operation that might fail temporarily

## The Code

```python
"""
Retry Pattern with Exponential Backoff
Handle transient failures with automatic retries and backoff strategies.
"""

from dataclasses import dataclass, field
from typing import Callable, Any, Optional, List, Type, Tuple
from datetime import datetime, timedelta
from enum import Enum, auto
import time
import random
import functools


class BackoffStrategy(Enum):
    EXPONENTIAL = auto()      # 2^n * base
    LINEAR = auto()           # n * base
    FIXED = auto()            # constant
    EXPONENTIAL_JITTER = auto()  # exp + random


@dataclass
class RetryConfig:
    """Configuration for retry behavior."""
    max_retries: int = 3
    base_delay: float = 1.0  # seconds
    max_delay: float = 60.0   # cap delay
    
    strategy: BackoffStrategy = BackoffStrategy.EXPONENTIAL_JITTER
    
    # Retry on these exceptions
    retry_on: Tuple[Type[Exception], ...] = (Exception,)
    
    # Don't retry these
    dont_retry_on: Tuple[Type[Exception], ...] = ()
    
    # Success condition
    should_retry: Optional[Callable[[Any], bool]] = None
    
    # Callbacks
    on_retry: Optional[Callable[[Exception, int, float], None]] = None
    on_success: Optional[Callable[[Any, int], None]] = None
    on_failure: Optional[Callable[[Exception, int], None]] = None


@dataclass
class RetryStats:
    """Statistics from retry operations."""
    attempts: int
    successes: int
    failures: int
    total_time: float
    avg_attempts: float
    

class RetryPolicy:
    """
    Configurable retry policy with exponential backoff.
    
    Usage:
        policy = RetryPolicy(
            max_retries=5,
            base_delay=2.0,
            strategy=BackoffStrategy.EXPONENTIAL_JITTER
        )
        
        result = policy.execute(fetch_api_data, url="https://api.example.com")
        
        # Or as decorator
        @policy.retry
        def fetch_data(url):
            return requests.get(url)
    """
    
    def __init__(self, config: Optional[RetryConfig] = None):
        self.config = config or RetryConfig()
        self.stats: List[RetryStats] = []
    
    def calculate_delay(self, attempt: int) -> float:
        """Calculate delay for retry attempt."""
        strat = self.config.strategy
        base = self.config.base_delay
        
        if strat == BackoffStrategy.FIXED:
            delay = base
        elif strat == BackoffStrategy.LINEAR:
            delay = base * attempt
        elif strat == BackoffStrategy.EXPONENTIAL:
            delay = base * (2 ** attempt)
        elif strat == BackoffStrategy.EXPONENTIAL_JITTER:
            delay = base * (2 ** attempt)
            # Add jitter (±25%)
            jitter = delay * 0.25
            delay += random.uniform(-jitter, jitter)
        else:
            delay = base
        
        return min(delay, self.config.max_delay)
    
    def should_retry_exception(self, exception: Exception) -> bool:
        """Check if exception should trigger retry."""
        # Check "don't retry" first
        for exc_type in self.config.dont_retry_on:
            if isinstance(exception, exc_type):
                return False
        
        # Check "do retry"
        for exc_type in self.config.retry_on:
            if isinstance(exception, exc_type):
                return True
        
        return False
    
    def execute(self, func: Callable, *args, **kwargs) -> Any:
        """Execute function with retries."""
        start_time = time.time()
        last_exception = None
        
        for attempt in range(self.config.max_retries + 1):
            try:
                result = func(*args, **kwargs)
                
                # Check custom success condition
                if self.config.should_retry and self.config.should_retry(result):
                    raise RetryRequiredException("Custom retry condition triggered")
                
                # Success!
                if self.config.on_success:
                    self.config.on_success(result, attempt + 1)
                
                return result
                
            except Exception as e:
                last_exception = e
                
                # Don't retry this exception
                if not self.should_retry_exception(e):
                    break
                
                # Max retries reached
                if attempt >= self.config.max_retries:
                    break
                
                # Calculate and wait
                delay = self.calculate_delay(attempt)
                
                if self.config.on_retry:
                    self.config.on_retry(e, attempt + 1, delay)
                
                time.sleep(delay)
        
        # All retries exhausted
        if self.config.on_failure:
            self.config.on_failure(last_exception, self.config.max_retries + 1)
        
        raise last_exception
    
    def retry(self, func: Callable) -> Callable:
        """Decorator for retry functionality."""
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            return self.execute(func, *args, **kwargs)
        return wrapper


class RetryRequiredException(Exception):
    """Raised when custom logic requires a retry."""
    pass

def example_basic_retry():
    """Basic retry with simulated failure."""
    print("=" * 70)
    print("Retry Pattern Demo - Basic Retry")
    print("=" * 70)
    
    class SimulatedAPI:
        def __init__(self, fail_count=2):
            self.fail_count = fail_count
            self.attempts = 0
        
        def fetch(self) -> str:
            self.attempts += 1
            if self.attempts <= self.fail_count:
                raise ConnectionError(f"Attempt {self.attempts} failed")
            return f"Success on attempt {self.attempts}"
    
    api = SimulatedAPI(fail_count=2)
    
    def on_retry(exc, attempt, delay):
        print(f"  Retry {attempt}: {exc.__class__.__name__} - waiting {delay:.2f}s")
    
    policy = RetryPolicy(
        RetryConfig(
            max_retries=3,
            base_delay=0.5,
            strategy=BackoffStrategy.EXPONENTIAL,
            on_retry=on_retry
        )
    )
    
    print(f"\nCalling API (will fail {api.fail_count} times)...")
    result = policy.execute(api.fetch)
    print(f"\n✅ Result: {result}")
    print(f"   Total attempts: {api.attempts}")


def example_decorator_usage():
    """Retry as decorator."""
    print("\n" + "=" * 70)
    print("Retry Pattern Demo - Decorator")
    print("=" * 70)
    
    policy = RetryPolicy(
        RetryConfig(
            max_retries=3,
            base_delay=0.3,
            strategy=BackoffStrategy.EXPONENTIAL_JITTER
        )
    )
    
    attempts = 0
    
    @policy.retry
    def unreliable_operation():
        nonlocal attempts
        attempts += 1
        if attempts < 3:
            raise TimeoutError(f"Timeout on attempt {attempts}")
        return f"Completed after {attempts} attempts"
    
    print(f"\nRunning unreliable operation...")
    result = unreliable_operation()
    print(f"✅ {result}")


def example_selective_retry():
    """Only retry specific exceptions."""
    print("\n" + "=" * 70)
    print("Retry Pattern Demo - Selective Retry")
    print("=" * 70)
    
    class DatabaseError(Exception):
        pass
    
    class ValidationError(Exception):
        pass
    
    policy = RetryPolicy(
        RetryConfig(
            max_retries=3,
            base_delay=0.5,
            retry_on=(ConnectionError, DatabaseError),
            dont_retry_on=(ValidationError,),
            on_retry=lambda e, a, d: print(f"  Retrying {e.__class__.__name__}...")
        )
    )
    
    print("\n--- Connection Error (will retry) ---")
    attempt = 0
    def network_call():
        nonlocal attempt
        attempt += 1
        if attempt < 2:
            raise ConnectionError("Network down")
        return "Network OK"
    
    result = policy.execute(network_call)
    print(f"✅ {result}")
    
    print("\n--- Validation Error (won't retry) ---")
    def invalid_call():
        raise ValidationError("Invalid input")
    
    try:
        policy.execute(invalid_call)
    except ValidationError as e:
        print(f"❌ {e} (not retried)")


def example_backoff_comparison():
    """Compare different backoff strategies."""
    print("\n" + "=" * 70)

    print("=" * 70)
    
    strategies = [
        BackoffStrategy.FIXED,
        BackoffStrategy.LINEAR,
        BackoffStrategy.EXPONENTIAL,
        BackoffStrategy.EXPONENTIAL_JITTER
    ]
    
    for strat in strategies:
        config = RetryConfig(
            max_retries=4,
            base_delay=1.0,
            max_delay=30.0,
            strategy=strat
        )
        policy = RetryPolicy(config)
        
        delays = []
        for attempt in range(5):
            delay = policy.calculate_delay(attempt)
            delays.append(delay)
        
        print(f"\n{strat.name}:")
        print(f"  Attempts 1-5 delays: {[f'{d:.2f}s' for d in delays]}")
        print(f"  Total wait: {sum(delays):.2f}s")


def example_custom_success_condition():
    """Retry based on return value."""

    # Retry if API returns None or empty

            base_delay=0.3,

    )

        nonlocal attempt
        attempt += 1
        if attempt < 3:

        return "Data loaded"
    
    print("\nFetching data that returns None initially...")
    result = policy.execute(flaky_api)
    print(f"✅ Got: '{result}' after {attempt} attempts")
```

## Backoff Strategy

| Strategy | Use Case |
|----------|----------|
| **FIXED** | Predictable intervals, simple services |
| **LINEAR** | Steady increase, rate limiting |
| **EXPONENTIAL** | Cascading failures, recovery needed |
| **EXP_JITTER** | Distributed systems, thundering herd |

## The Math

```
EXPONENTIAL:     delay = base × 2^attempt
LINEAR:          delay = base × attempt
FIXED:           delay = base
EXP_JITTER:      delay = (base × 2^attempt) ± 25%

Example (base=1s, max=30s):
Attempt 0: 1s  → 1s
Attempt 1: 2s  → 2s  
Attempt 2: 4s  → 4s
Attempt 3: 8s  → 8s
Attempt 4: 16s → 16s
Attempt 5: 32s → capped at 30s

Total: ~61 seconds over 6 attempts
```

## Why This Matters

- **Networks are flaky** — Retries hide transient failures
- **Costs accumulate** — Don't retry endlessly
- **Backoff prevents overload** — Exponential gives services time to recover  
- **Jitter prevents herds** — Randomness stops synchronized retries
- **Be selective** — Only retry on expected exceptions

**Retry intelligently—fail fast on logic errors, retry patiently on network issues.**
