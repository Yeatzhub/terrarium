# PnL Tracker — Realized, Unrealized & Performance Metrics

**Purpose:** Track trading performance with full accounting of realized/unrealized PnL, drawdowns, and expectancy  
**Use Case:** Performance analysis, tax reporting, strategy evaluation

## The Code

```python
"""
PnL Tracker
Comprehensive profit/loss tracking with drawdown analysis, win rates,
expectancy calculations, and performance reporting.
"""

from dataclasses import dataclass, field
from typing import List, Dict, Optional, Tuple
from datetime import datetime, timedelta
from enum import Enum, auto
from collections import defaultdict
import statistics


class TradeStatus(Enum):
    OPEN = auto()
    CLOSED = auto()
    PARTIAL = auto()


@dataclass
class Trade:
    """Individual trade record."""
    id: str
    symbol: str
    entry_date: datetime
    entry_price: float
    quantity: float
    side: str  # "long" or "short"
    
    exit_date: Optional[datetime] = None
    exit_price: Optional[float] = None
    status: TradeStatus = TradeStatus.OPEN
    
    commissions: float = 0.0
    fees: float = 0.0
    
    # Partial exits
    partial_exits: List[Tuple[float, float, datetime]] = field(default_factory=list)  # (qty, price, date)
    
    @property
    def position_value(self) -> float:
        return self.quantity * self.entry_price
    
    @property
    def realized_pnl(self) -> float:
        """Realized PnL from closed portion."""
        if not self.partial_exits:
            return 0.0
        
        total = 0.0
        for qty, price, _ in self.partial_exits:
            if self.side == "long":
                total += qty * (price - self.entry_price)
            else:
                total += qty * (self.entry_price - price)
        
        return total - self.commissions - self.fees
    
    @property
    def unrealized_pnl(self, current_price: Optional[float] = None) -> float:
        """Unrealized PnL on remaining position."""
        if not current_price or self.status == TradeStatus.CLOSED:
            return 0.0
        
        remaining_qty = self.quantity - sum(qty for qty, _, _ in self.partial_exits)
        if remaining_qty <= 0:
            return 0.0
        
        if self.side == "long":
            return remaining_qty * (current_price - self.entry_price)
        else:
            return remaining_qty * (self.entry_price - current_price)
    
    def close(self, exit_price: float, exit_date: datetime, qty: Optional[float] = None):
        """Close trade or portion of trade."""
        qty = qty or self.quantity
        
        if qty >= self.remaining_quantity:
            self.exit_price = exit_price
            self.exit_date = exit_date
            self.status = TradeStatus.CLOSED
        else:
            self.partial_exits.append((qty, exit_price, exit_date))
            if sum(q for q, _, _ in self.partial_exits) >= self.quantity:
                self.status = TradeStatus.CLOSED
            else:
                self.status = TradeStatus.PARTIAL
    
    @property
    def remaining_quantity(self) -> float:
        return self.quantity - sum(qty for qty, _, _ in self.partial_exits)


@dataclass
class DailyPnL:
    """Daily PnL summary."""
    date: datetime
    realized_pnl: float
    unrealized_pnl: float
    num_trades: int
    num_winners: int
    num_losers: int
    fees: float
    
    @property
    def net_pnl(self) -> float:
        return self.realized_pnl + self.unrealized_pnl - self.fees
    
    @property
    def win_rate(self) -> float:
        if self.num_trades == 0:
            return 0.0
        return self.num_winners / self.num_trades


@dataclass
class PerformanceMetrics:
    """Overall performance statistics."""
    total_realized_pnl: float
    total_unrealized_pnl: float
    gross_pnl: float
    total_fees: float
    net_pnl: float
    
    num_trades: int
    num_winners: int
    num_losers: int
    win_rate: float
    
    avg_winner: float
    avg_loser: float
    avg_trade: float
    
    profit_factor: float  # Gross wins / Gross losses
    expectancy: float  # Expected value per trade
    expectancy_percent: float  # Expected return per dollar risked
    
    max_drawdown_pct: float
    current_drawdown_pct: float
    max_drawdown_dollar: float
    
    sharpe_estimate: float  # Simplified Sharpe
    consecutive_wins: int
    consecutive_losses: int
    
    best_day: float
    worst_day: float
    avg_daily_pnl: float


@dataclass
class DrawdownPeriod:
    """Individual drawdown period."""
    start_date: datetime
    end_date: Optional[datetime]
    peak_equity: float
    trough_equity: float
    drawdown_pct: float
    duration_days: int
    recovered: bool


class PnLTracker:
    """
    Comprehensive PnL tracking system.
    
    Usage:
        tracker = PnLTracker()
        
        # Record entry
        tracker.enter_trade("AAPL", 150.0, 100, "long", date=datetime.now())
        
        # Update with current price for unrealized PnL
        tracker.update_price("AAPL", 155.0, datetime.now())
        
        # Record exit
        tracker.exit_trade("AAPL", 160.0, datetime.now())
        
        # Get report
        metrics = tracker.get_metrics()
        print(f"Win rate: {metrics.win_rate:.1%}")
    """
    
    def __init__(self, initial_capital: float = 100000.0):
        self.initial_capital = initial_capital
        self.trades: Dict[str, Trade] = {}
        self.closed_trades: List[Trade] = []
        self.current_prices: Dict[str, float] = {}
        self.daily_pnl: Dict[str, DailyPnL] = {}
        
        self.equity_curve: List[Tuple[datetime, float]] = []
        self.peak_equity = initial_capital
        self.current_equity = initial_capital
        self.drawdowns: List[DrawdownPeriod] = []
        self.active_drawdown: Optional[DrawdownPeriod] = None
        
        self._trade_counter = 0
    
    def enter_trade(
        self,
        symbol: str,
        entry_price: float,
        quantity: float,
        side: str = "long",
        date: Optional[datetime] = None,
        commissions: float = 0.0,
        fees: float = 0.0,
        trade_id: Optional[str] = None
    ) -> str:
        """Record new trade entry."""
        self._trade_counter += 1
        trade_id = trade_id or f"{symbol}_{self._trade_counter}_{date or datetime.now()}"
        
        trade = Trade(
            id=trade_id,
            symbol=symbol,
            entry_date=date or datetime.now(),
            entry_price=entry_price,
            quantity=quantity,
            side=side,
            commissions=commissions,
            fees=fees
        )
        
        self.trades[trade_id] = trade
        return trade_id
    
    def update_price(self, symbol: str, price: float, date: Optional[datetime] = None):
        """Update current market price for unrealized PnL calculation."""
        self.current_prices[symbol] = price
        self._recalculate_equity(date or datetime.now())
    
    def exit_trade(
        self,
        trade_id: str,
        exit_price: float,
        date: Optional[datetime] = None,
        qty: Optional[float] = None
    ) -> float:
        """Record trade exit, return realized PnL."""
        if trade_id not in self.trades:
            raise ValueError(f"Trade {trade_id} not found")
        
        trade = self.trades[trade_id]
        exit_date = date or datetime.now()
        
        # Calculate realized PnL before closing
        if qty and qty < trade.remaining_quantity:
            # Partial close
            if trade.side == "long":
                realized = qty * (exit_price - trade.entry_price)
            else:
                realized = qty * (trade.entry_price - exit_price)
        else:
            # Full close
            realized = trade.realized_pnl
            if trade.side == "long":
                realized += trade.remaining_quantity * (exit_price - trade.entry_price)
            else:
                realized += trade.remaining_quantity * (trade.entry_price - exit_price)
            realized -= trade.commissions + trade.fees
        
        trade.close(exit_price, exit_date, qty)
        
        # Record daily PnL
        self._record_daily_pnl(exit_date, realized, trade.fees)
        
        if trade.status == TradeStatus.CLOSED:
            self.closed_trades.append(trade)
            del self.trades[trade_id]
        
        # Update equity
        self._recalculate_equity(exit_date)
        
        return realized
    
    def _record_daily_pnl(self, date: datetime, realized: float, fees: float):
        """Record PnL for daily tracking."""
        date_key = date.strftime("%Y-%m-%d")
        
        if date_key not in self.daily_pnl:
            self.daily_pnl[date_key] = DailyPnL(
                date=date,
                realized_pnl=0.0,
                unrealized_pnl=0.0,
                num_trades=0,
                num_winners=0,
                num_losers=0,
                fees=0.0
            )
        
        daily = self.daily_pnl[date_key]
        daily.realized_pnl += realized
        daily.fees += fees
        daily.num_trades += 1
        
        if realized > 0:
            daily.num_winners += 1
        elif realized < 0:
            daily.num_losers += 1
    
    def _recalculate_equity(self, date: datetime):
        """Recalculate current equity and drawdown."""
        # Start with initial + all realized PnL
        equity = self.initial_capital
        
        # Add realized from closed trades
        for trade in self.closed_trades:
            # Calculate total realized from partials + full close
            total_realized = trade.realized_pnl
            if trade.side == "long" and trade.exit_price:
                remaining = trade.remaining_quantity
                total_realized += remaining * (trade.exit_price - trade.entry_price)
            elif trade.side == "short" and trade.exit_price:
                remaining = trade.remaining_quantity
                total_realized += remaining * (trade.entry_price - trade.exit_price)
            equity += total_realized - trade.commissions - trade.fees
        
        # Add unrealized from open trades
        for trade in self.trades.values():
            current_price = self.current_prices.get(trade.symbol)
            if current_price:
                unrealized = trade.unrealized_pnl(current_price)
                equity += trade.realized_pnl + unrealized - trade.commissions - trade.fees
            else:
                # No current price, use entry price (no unrealized PnL)
                equity += trade.realized_pnl - trade.commissions - trade.fees
        
        self.current_equity = equity
        self.equity_curve.append((date, equity))
        
        # Update peak and drawdown
        if equity > self.peak_equity:
            self.peak_equity = equity
            # Close active drawdown if exists
            if self.active_drawdown:
                self.active_drawdown.end_date = date
                self.active_drawdown.recovered = True
                self.active_drawdown.duration_days = (
                    date - self.active_drawdown.start_date
                ).days
                self.drawdowns.append(self.active_drawdown)
                self.active_drawdown = None
        else:
            # Check for new or continuing drawdown
            drawdown_pct = (self.peak_equity - equity) / self.peak_equity * 100
            
            if drawdown_pct > 0:
                if not self.active_drawdown:
                    self.active_drawdown = DrawdownPeriod(
                        start_date=date,
                        end_date=None,
                        peak_equity=self.peak_equity,
                        trough_equity=equity,
                        drawdown_pct=drawdown_pct,
                        duration_days=0,
                        recovered=False
                    )
                else:
                    # Update trough if lower
                    if equity < self.active_drawdown.trough_equity:
                        self.active_drawdown.trough_equity = equity
                        self.active_drawdown.drawdown_pct = drawdown_pct
    
    def get_metrics(self) -> PerformanceMetrics:
        """Calculate comprehensive performance metrics."""
        if not self.closed_trades:
            return PerformanceMetrics(
                total_realized_pnl=0.0,
                total_unrealized_pnl=0.0,
                gross_pnl=0.0,
                total_fees=0.0,
                net_pnl=0.0,
                num_trades=0,
                num_winners=0,
                num_losers=0,
                win_rate=0.0,
                avg_winner=0.0,
                avg_loser=0.0,
                avg_trade=0.0,
                profit_factor=0.0,
                expectancy=0.0,
                expectancy_percent=0.0,
                max_drawdown_pct=0.0,
                current_drawdown_pct=0.0,
                max_drawdown_dollar=0.0,
                sharpe_estimate=0.0,
                consecutive_wins=0,
                consecutive_losses=0,
                best_day=0.0,
                worst_day=0.0,
                avg_daily_pnl=0.0
            )
        
        # Trade statistics
        winners = []
        losers = []
        total_fees = 0.0
        
        for trade in self.closed_trades:
            # Calculate total PnL for this trade
            pnl = trade.realized_pnl
            if trade.exit_price:
                remaining = trade.remaining_quantity
                if trade.side == "long":
                    pnl += remaining * (trade.exit_price - trade.entry_price)
                else:
                    pnl += remaining * (trade.entry_price - trade.exit_price)
            pnl -= trade.commissions + trade.fees
            
            if pnl > 0:
                winners.append(pnl)
            else:
                losers.append(abs(pnl))
            
            total_fees += trade.commissions + trade.fees
        
        # Include partial realized PnL from open trades
        for trade in self.trades.values():
            total_fees += trade.commissions + trade.fees
        
        num_winners = len(winners)
        num_losers = len(losers)
        num_trades = num_winners + num_losers
        
        total_wins = sum(winners) if winners else 0
        total_losses = sum(losers) if losers else 0
        
        avg_winner = total_wins / num_winners if num_winners > 0 else 0
        avg_loser = total_losses / num_losers if num_losers > 0 else 0
        
        win_rate = num_winners / num_trades if num_trades > 0 else 0
        
        # Profit factor
        profit_factor = total_wins / total_losses if total_losses > 0 else float('inf')
        
        # Expectancy
        avg_trade = (total_wins - total_losses) / num_trades if num_trades > 0 else 0
        expectancy = (win_rate * avg_winner) - ((1 - win_rate) * avg_loser)
        expectancy_percent = (expectancy / self.initial_capital) * 100 if self.initial_capital > 0 else 0
        
        # Drawdown analysis
        max_dd_pct = 0.0
        max_dd_dollar = 0.0
        for dd in self.drawdowns:
            if dd.drawdown_pct > max_dd_pct:
                max_dd_pct = dd.drawdown_pct
                max_dd_dollar = dd.peak_equity - dd.trough_equity
        
        if self.active_drawdown and self.active_drawdown.drawdown_pct > max_dd_pct:
            max_dd_pct = self.active_drawdown.drawdown_pct
            max_dd_dollar = self.active_drawdown.peak_equity - self.active_drawdown.trough_equity
        
        current_dd_pct = 0.0
        if self.peak_equity > 0:
            current_dd_pct = (self.peak_equity - self.current_equity) / self.peak_equity * 100
        
        # Consecutive wins/losses
        consec_wins = 0
        consec_losses = 0
        max_consec_wins = 0
        max_consec_losses = 0
        
        all_trades_pnl = []
        for trade in sorted(self.closed_trades, key=lambda t: t.exit_date or t.entry_date):
            pnl = trade.realized_pnl
            if trade.exit_price:
                remaining = trade.remaining_quantity
                if trade.side == "long":
                    pnl += remaining * (trade.exit_price - trade.entry_price)
                else:
                    pnl += remaining * (trade.entry_price - trade.exit_price)
            pnl -= trade.commissions + trade.fees
            all_trades_pnl.append(pnl)
        
        for pnl in all_trades_pnl:
            if pnl > 0:
                consec_wins += 1
                consec_losses = 0
                max_consec_wins = max(max_consec_wins, consec_wins)
            else:
                consec_losses += 1
                consec_wins = 0
                max_consec_losses = max(max_consec_losses, consec_losses)
        
        # Daily stats
        daily_pnls = [d.net_pnl for d in self.daily_pnl.values()]
        best_day = max(daily_pnls) if daily_pnls else 0
        worst_day = min(daily_pnls) if daily_pnls else 0
        avg_daily = sum(daily_pnls) / len(daily_pnls) if daily_pnls else 0
        
        # Simple Sharpe estimate (assuming 252 trading days, 2% risk-free)
        if len(daily_pnls) > 1:
            try:
                volatility = statistics.stdev(daily_pnls) * math.sqrt(252)
                sharpe = (avg_daily * 252 - 0.02) / volatility if volatility > 0 else 0
            except:
                sharpe = 0.0
        else:
            sharpe = 0.0
        
        # Unrealized PnL
        unrealized = 0.0
        for trade in self.trades.values():
            current_price = self.current_prices.get(trade.symbol)
            if current_price:
                unrealized += trade.unrealized_pnl(current_price)
        
        total_realized = sum(winners) - sum(losers) - total_fees
        
        return PerformanceMetrics(
            total_realized_pnl=total_realized,
            total_unrealized_pnl=unrealized,
            gross_pnl=total_realized + unrealized,
            total_fees=total_fees,
            net_pnl=total_realized + unrealized,
            num_trades=num_trades,
            num_winners=num_winners,
            num_losers=num_losers,
            win_rate=win_rate,
            avg_winner=avg_winner,
            avg_loser=avg_loser,
            avg_trade=avg_trade,
            profit_factor=profit_factor,
            expectancy=expectancy,
            expectancy_percent=expectancy_percent,
            max_drawdown_pct=max_dd_pct,
            current_drawdown_pct=current_dd_pct,
            max_drawdown_dollar=max_dd_dollar,
            sharpe_estimate=sharpe,
            consecutive_wins=max_consec_wins,
            consecutive_losses=max_consec_losses,
            best_day=best_day,
            worst_day=worst_day,
            avg_daily_pnl=avg_daily
        )
    
    def get_daily_report(self, days: int = 30) -> List[DailyPnL]:
        """Get recent daily PnL."""
        sorted_days = sorted(self.daily_pnl.values(), key=lambda d: d.date, reverse=True)
        return sorted_days[:days]
    
    def get_drawdown_report(self) -> List[DrawdownPeriod]:
        """Get all drawdown periods."""
        all_dd = list(self.drawdowns)
        if self.active_drawdown:
            all_dd.append(self.active_drawdown)
        return sorted(all_dd, key=lambda d: d.drawdown_pct, reverse=True)


import math


def format_metrics(metrics: PerformanceMetrics) -> str:
    """Pretty print metrics."""
    lines = [
        "=" * 60,
        "PERFORMANCE REPORT",
        "=" * 60,
        "",
        f"Total Trades:     {metrics.num_trades}",
        f"Winners:          {metrics.num_winners} ({metrics.win_rate:.1%})",
        f"Losers:           {metrics.num_losers} ({1-metrics.win_rate:.1%})",
        "",
        f"Gross PnL:        ${metrics.gross_pnl:+,.2f}",
        f"Total Fees:       ${metrics.total_fees:,.2f}",
        f"Net PnL:          ${metrics.net_pnl:+,.2f}",
        f"Unrealized:       ${metrics.total_unrealized_pnl:+,.2f}",
        "",
        f"Avg Winner:       ${metrics.avg_winner:,.2f}",
        f"Avg Loser:        ${metrics.avg_loser:,.2f}",
        f"Avg Trade:        ${metrics.avg_trade:+,.2f}",
        "",
        f"Profit Factor:    {metrics.profit_factor:.2f}",
        f"Expectancy:       ${metrics.expectancy:+,.2f} per trade",
        f"Expectancy %:     {metrics.expectancy_percent:+.3f}%",
        "",
        f"Max Drawdown:     {metrics.max_drawdown_pct:.1f}% (${metrics.max_drawdown_dollar:,.2f})",
        f"Current DD:       {metrics.current_drawdown_pct:.1f}%",
        "",
        f"Consecutive Wins:  {metrics.consecutive_wins}",
        f"Consecutive Loss: {metrics.consecutive_losses}",
        "",
        f"Best Day:         ${metrics.best_day:+,.2f}",
        f"Worst Day:        ${metrics.worst_day:+,.2f}",
        f"Avg Daily:        ${metrics.avg_daily_pnl:+,.2f}",
        f"Sharpe (est):     {metrics.sharpe_estimate:.2f}",
        "=" * 60,
    ]
    return "\n".join(lines)

Profit Factor: Gross wins / Gross losses (2.0+ is good)
Expectancy: Expected profit per trade (positive = edge)
Max Drawdown: Worst peak-to-trough decline (keep < 20%)
Sharpe Ratio: Risk-adjusted returns (>1.0 is decent, >2.0 is excellent)

```

## Key Metrics Guide

| Metric | Good | Great | Formula |
|--------|------|-------|---------|
| **Win Rate** | 40-50% | 55%+ | Winners / Total |
| **Profit Factor** | 1.5+ | 2.0+ | Gross Wins / Gross Losses |
| **Expectancy** | > $0 | > $100/trade | (Win% × Avg Win) - (Loss% × Avg Loss) |
| **Max Drawdown** | < 15% | < 10% | Peak - Trough |
| **Sharpe** | > 1.0 | > 2.0 | Return / Volatility |
| **Avg Trade** | > $0 | > 0.5% of capital | Net PnL / Trades |

## Why Track Everything

- **Win rate alone is useless** — 90% win rate with tiny wins and huge losses loses money
- **Expectancy is the truth** — Positive expectancy = you have an edge
- **Drawdowns kill traders** — 50% drawdown requires 100% gain to recover
- **Fees matter** — High-frequency with small edge gets eaten by fees
- **You can't improve what you don't measure** — Track everything
