## Bug Fixed

Original used `heapq` for bids but didn't negate prices - resulted in **wrong ordering**. 
Bids need **max-heap** (highest price first), asks need **min-heap** (lowest price first).

```python
"""
Order Book Pattern v2
Fixed heap implementation + enhancements.
"""

from dataclasses import dataclass, field
from decimal import Decimal
from typing import Optional
from collections import defaultdict
import heapq
from enum import Enum
from datetime import datetime


class Side(Enum):
    BUY = "buy"
    SELL = "sell"


@dataclass
class Order:
    """Order with proper comparison for heap."""
    price: Decimal
    timestamp: float
    order_id: str = field(compare=False)
    side: Side = field(compare=False)
    size: Decimal = field(compare=False)
    filled: Decimal = field(default=Decimal("0"), compare=False)
    
    @property
    def remaining(self) -> Decimal:
        return self.size - self.filled
    
    @property
    def is_filled(self) -> bool:
        return self.remaining <= 0
    
    def heap_key(self) -> tuple:
        """Return comparison key for heap insertion."""
        if self.side == Side.BUY:
            # Bids: negate price for max-heap behavior
            return (-float(self.price), self.timestamp)
        else:
            # Asks: normal min-heap
            return (float(self.price), self.timestamp)


@dataclass
class Trade:
    """Executed trade."""
    taker_order_id: str
    maker_order_id: str
    price: Decimal
    size: Decimal
    side: Side  # Taker side
    timestamp: datetime = field(default_factory=datetime.utcnow)


class OrderBook:
    """
    Fixed order book with proper heap ordering.
    
    Bids: Max-heap via negated prices (best/highest bid first)
    Asks: Min-heap (best/lowest ask first)
    Same price: FIFO (timestamp order)
    """
    
    def __init__(self, symbol: str):
        self.symbol = symbol
        self._bid_heap: list[tuple] = []  # (heap_key, order)
        self._ask_heap: list[tuple] = []  # (heap_key, order)
        self.orders: dict[str, Order] = {}
        self.trades: list[Trade] = []
        self._order_counter = 0
        self._trade_volume = Decimal("0")
        self._trade_value = Decimal("0")
    
    def _next_order_id(self) -> str:
        self._order_counter += 1
        return f"{self.symbol}-{self._order_counter:06d}"
    
    def _push_order(self, order: Order) -> None:
        """Push order to correct heap with proper key."""
        heap = self._bid_heap if order.side == Side.BUY else self._ask_heap
        heapq.heappush(heap, (order.heap_key(), order))
    
    def _peek_best(self, side: Side) -> Optional[Order]:
        """Peek at best order for side, skipping filled orders."""
        heap = self._bid_heap if side == Side.BUY else self._ask_heap
        
        while heap:
            _, order = heap[0]
            if order.order_id in self.orders and not order.is_filled:
                return order
            heapq.heappop(heap)  # Remove stale/filled
        return None
    
    def _pop_best(self, side: Side) -> Optional[Order]:
        """Pop best order for side."""
        heap = self._bid_heap if side == Side.BUY else self._ask_heap
        
        while heap:
            _, order = heapq.heappop(heap)
            if order.order_id in self.orders and not order.is_filled:
                return order
        return None
    
    def _clean_heap(self, side: Side) -> None:
        """Remove filled/cancelled orders from heap."""
        heap = self._bid_heap if side == Side.BUY else self._ask_heap
        valid = [(k, o) for k, o in heap if o.order_id in self.orders and not o.is_filled]
        heap.clear()
        heap.extend(valid)
        heapq.heapify(heap)
    
    def add_order(
        self,
        side: Side,
        price: Decimal,
        size: Decimal,
        order_id: Optional[str] = None
    ) -> tuple[str, list[Trade]]:
        """Add order, match if possible, rest goes to book."""
        order_id = order_id or self._next_order_id()
        timestamp = datetime.utcnow().timestamp()
        
        order = Order(
            price=price,
            timestamp=timestamp,
            order_id=order_id,
            side=side,
            size=size
        )
        
        trades = self._match_order(order)
        
        if not order.is_filled:
            self.orders[order_id] = order
            self._push_order(order)
        
        return order_id, trades
    
    def _match_order(self, order: Order) -> list[Trade]:
        """Match taker order against book."""
        trades = []
        opposite = Side.SELL if order.side == Side.BUY else Side.BUY
        
        while not order.is_filled:
            maker = self._peek_best(opposite)
            if not maker:
                break
            
            # Check price crosses
            if order.side == Side.BUY:
                if maker.price > order.price:
                    break
            else:
                if maker.price < order.price:
                    break
            
            # Execute trade
            maker = self._pop_best(opposite)
            trade_size = min(order.remaining, maker.remaining)
            
            trade = Trade(
                taker_order_id=order.order_id,
                maker_order_id=maker.order_id,
                price=maker.price,
                size=trade_size,
                side=order.side
            )
            trades.append(trade)
            
            # Update fills
            order.filled += trade_size
            maker.filled += trade_size
            
            # Track stats
            self._trade_volume += trade_size
            self._trade_value += trade_size * maker.price
            
            # Re-add maker if partially filled
            if not maker.is_filled:
                self._push_order(maker)
            else:
                del self.orders[maker.order_id]
        
        self.trades.extend(trades)
        return trades
    
    def cancel_order(self, order_id: str) -> bool:
        """Cancel order (lazy removal from heap)."""
        if order_id not in self.orders:
            return False
        del self.orders[order_id]
        return True
    
    @property
    def best_bid(self) -> Optional[tuple[Decimal, Decimal]]:
        """(price, size) of best bid."""
        order = self._peek_best(Side.BUY)
        return (order.price, order.remaining) if order else None
    
    @property
    def best_ask(self) -> Optional[tuple[Decimal, Decimal]]:
        """(price, size) of best ask."""
        order = self._peek_best(Side.SELL)
        return (order.price, order.remaining) if order else None
    
    @property
    def spread(self) -> Optional[Decimal]:
        """Bid-ask spread."""
        bid = self.best_bid
        ask = self.best_ask
        return ask[0] - bid[0] if bid and ask else None
    
    @property
    def mid_price(self) -> Optional[Decimal]:
        """Mid price."""
        bid = self.best_bid
        ask = self.best_ask
        return (bid[0] + ask[0]) / 2 if bid and ask else None
    
    @property
    def vwap(self) -> Optional[Decimal]:
        """Volume-weighted average price of trades."""
        return self._trade_value / self._trade_volume if self._trade_volume > 0 else None
    
    @property
    def stats(self) -> dict:
        """Trading statistics."""
        return {
            "trades": len(self.trades),
            "volume": float(self._trade_volume),
            "value": float(self._trade_value),
            "vwap": float(self.vwap) if self.vwap else None,
            "open_orders": len(self.orders)
        }
    
    def get_depth(self, levels: int = 5) -> dict:
        """Get aggregated depth by price level."""
        bid_levels = defaultdict(Decimal)
        ask_levels = defaultdict(Decimal)
        
        for order in self.orders.values():
            if order.remaining <= 0:
                continue
            if order.side == Side.BUY:
                bid_levels[order.price] += order.remaining
            else:
                ask_levels[order.price] += order.remaining
        
        return {
            "bids": sorted(bid_levels.items(), key=lambda x: x[0], reverse=True)[:levels],
            "asks": sorted(ask_levels.items(), key=lambda x: x[0])[:levels]
        }
    
    def __str__(self) -> str:
        """Pretty print."""
        depth = self.get_depth(5)
        lines = [f"\n{'='*11} {self.symbol} {'='*11}"]
        lines.append(f"Spread: {self.spread} | Mid: {self.mid_price} | VWAP: {self.vwap}")
        lines.append(f"\n{'ASKS':>22}")
        for price, size in reversed(depth["asks"]):
            lines.append(f"{price:>12} | {size:>8}")
        lines.append("-" * 23)
        for price, size in depth["bids"]:
            lines.append(f"{price:>12} | {size:>8}")
        lines.append(f"{'BIDS':>22}")
        lines.append(f"\nStats: {self.stats}")
        return "\n".join(lines)


# === Usage ===

def demo():
    book = OrderBook("BTCUSD")
    
    # Add liquidity
    for price in [50020, 50010, 50000]:
        book.add_order(Side.SELL, Decimal(str(price)), Decimal("1.0"))
    
    for price in [49990, 49980, 49970]:
        book.add_order(Side.BUY, Decimal(str(price)), Decimal("1.5"))
    
    print(book)
    
    # Crossing order (market buy)
    print("\n--- Market Buy 2.0 @ 50010 ---")
    _, trades = book.add_order(Side.BUY, Decimal("50010"), Decimal("2.0"))
    print(f"Trades: {len(trades)}")
    for t in trades:
        print(f"  {t.size} @ {t.price}")
    
    print(book)


if __name__ == "__main__":
    demo()
```

## Key Fixes

| Issue | Fix |
|-------|-----|
| Bid heap wrong order | Negate price in `heap_key()` for max-heap |
| Stale orders in heap | `_peek_best()` skips filled/cancelled |
| No trade stats | Added VWAP, volume, trade count |

## Heap Comparison Logic

```python
# Bids: -price means higher price = smaller key = heap top
# Asks: +price means lower price = smaller key = heap top

def heap_key(self) -> tuple:
    if self.side == Side.BUY:
        return (-float(self.price), self.timestamp)  # Max-heap
    else:
        return (float(self.price), self.timestamp)   # Min-heap
```
