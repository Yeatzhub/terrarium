# Monte Carlo Backtest Validator
*Ghost Learning | 2026-02-21*

Validate strategy robustness through Monte Carlo simulation. Randomizes trade sequences to test drawdown distribution and confidence intervals. Essential for knowing if your backtest results are luck or skill.

```python
#!/usr/bin/env python3
"""
Monte Carlo Backtest Validator
Validates strategy robustness via Monte Carlo simulation.

Usage:
    python mc_validator.py trades.csv --simulations 10000
    python mc_validator.py trades.csv --randomize --bootstrap --equity 10000
"""

import argparse
import csv
import json
import random
from dataclasses import dataclass, field
from datetime import datetime
from decimal import Decimal
from pathlib import Path
from typing import Optional, list
from statistics import mean, stdev
import math


@dataclass
class Trade:
    """Trade record."""
    pnl: Decimal
    return_pct: Decimal
    timestamp: Optional[datetime] = None
    
    @classmethod
    def from_csv_row(cls, row: dict) -> "Trade":
        """Parse trade from CSV row."""
        pnl = Decimal(row.get("pnl", row.get("profit", "0")))
        ret_pct = Decimal(row.get("return_pct", row.get("return", "0")))
        ts = None
        if "timestamp" in row:
            ts = datetime.fromisoformat(row["timestamp"])
        return cls(pnl=pnl, return_pct=ret_pct, timestamp=ts)


@dataclass
class SimulationResult:
    """Single Monte Carlo simulation outcome."""
    final_equity: Decimal
    total_return: Decimal
    max_drawdown: Decimal
    max_drawdown_pct: Decimal
    win_rate: Decimal
    profit_factor: Decimal
    sharpe: Decimal


@dataclass
class MCAnalysis:
    """Aggregated Monte Carlo results."""
    simulations: int
    
    # Return distribution
    median_return: Decimal
    mean_return: Decimal
    return_5th: Decimal
    return_95th: Decimal
    return_std: Decimal
    
    # Drawdown distribution
    median_dd: Decimal
    max_dd: Decimal
    dd_95th: Decimal
    dd_99th: Decimal
    
    # Probability metrics
    prob_profit: float           # Probability of profitable outcome
    prob_dd_over_20: float       # Probability of DD > 20%
    prob_dd_over_30: float       # Probability of DD > 30%
    
    # Confidence interval
    ci_return: tuple[float, float]  # 90% CI for return
    ci_dd: tuple[float, float]      # 90% CI for DD
    
    # Worst case scenarios
    worst_return: Decimal
    best_return: Decimal
    worst_dd: Decimal
    
    def to_dict(self) -> dict:
        return {
            "simulations": self.simulations,
            "return_distribution": {
                "median": float(self.median_return),
                "mean": float(self.mean_return),
                "p5": float(self.return_5th),
                "p95": float(self.return_95th),
                "std": float(self.return_std)
            },
            "drawdown_distribution": {
                "median": float(self.median_dd),
                "max": float(self.max_dd),
                "p95": float(self.dd_95th),
                "p99": float(self.dd_99th)
            },
            "probabilities": {
                "profitable": self.prob_profit,
                "dd_over_20pct": self.prob_dd_over_20,
                "dd_over_30pct": self.prob_dd_over_30
            },
            "confidence_intervals": {
                "return_90pct": self.ci_return,
                "dd_90pct": self.ci_dd
            },
            "worst_case": {
                "worst_return": float(self.worst_return),
                "best_return": float(self.best_return),
                "worst_dd": float(self.worst_dd)
            }
        }


class MCValidator:
    """Monte Carlo simulation for backtest validation."""
    
    def __init__(self, seed: Optional[int] = None):
        self.seed = seed
        if seed:
            random.seed(seed)
    
    def shuffle_trades(self, trades: list[Trade]) -> list[Trade]:
        """Randomize trade order."""
        shuffled = trades.copy()
        random.shuffle(shuffled)
        return shuffled
    
    def bootstrap_sample(self, trades: list[Trade], n: Optional[int] = None) -> list[Trade]:
        """Sample with replacement (bootstrap)."""
        n = n or len(trades)
        samples = [random.choice(trades) for _ in range(n)]
        return samples
    
    def simulate_equity_curve(
        self, 
        trades: list[Trade], 
        start_equity: Decimal = Decimal("10000"),
        shuffle: bool = True
    ) -> SimulationResult:
        """Run single simulation and generate equity curve."""
        if shuffle:
            trades = self.shuffle_trades(trades)
        
        equity = start_equity
        max_equity = equity
        max_dd = Decimal("0")
        total_wins = Decimal("0")
        total_losses = Decimal("0")
        wins = 0
        losses = 0
        equity_curve = [equity]
        returns = []
        
        for trade in trades:
            equity += trade.pnl
            equity_curve.append(equity)
            
            if equity > max_equity:
                max_equity = equity
            
            dd = equity - max_equity
            if dd < max_dd:
                max_dd = dd
            
            if trade.pnl > 0:
                total_wins += trade.pnl
                wins += 1
            else:
                total_losses += abs(trade.pnl)
                losses += 1
            
            if start_equity > 0:
                returns.append(float(trade.pnl / start_equity))
        
        total_trades = wins + losses
        win_rate = Decimal(wins) / Decimal(total_trades) if total_trades > 0 else Decimal("0")
        
        profit_factor = total_wins / total_losses if total_losses > 0 else total_wins
        
        max_dd_pct = (abs(max_dd) / start_equity * 100) if start_equity > 0 else Decimal("0")
        total_return = ((equity - start_equity) / start_equity * 100) if start_equity > 0 else Decimal("0")
        
        # Sharpe-like ratio
        if returns and len(returns) > 1:
            avg_ret = mean(returns)
            ret_std = stdev(returns) if len(returns) > 1 else 0.001
            sharpe = Decimal(str((avg_ret / ret_std) * math.sqrt(252))) if ret_std > 0 else Decimal("0")
        else:
            sharpe = Decimal("0")
        
        return SimulationResult(
            final_equity=equity,
            total_return=total_return,
            max_drawdown=abs(max_dd),
            max_drawdown_pct=max_dd_pct,
            win_rate=win_rate * 100,
            profit_factor=profit_factor,
            sharpe=sharpe
        )
    
    def run(
        self,
        trades: list[Trade],
        start_equity: Decimal = Decimal("10000"),
        simulations: int = 10000,
        method: str = "shuffle"  # 'shuffle' or 'bootstrap'
    ) -> MCAnalysis:
        """Run full Monte Carlo analysis."""
        results: list[SimulationResult] = []
        
        for _ in range(simulations):
            sim_trades = trades if method == "shuffle" else self.bootstrap_sample(trades, len(trades))
            result = self.simulate_equity_curve(sim_trades, start_equity, shuffle=(method == "shuffle"))
            results.append(result)
        
        return self._analyze(results)
    
    def _analyze(self, results: list[SimulationResult]) -> MCAnalysis:
        """Aggregate simulation results."""
        returns = [r.total_return for r in results]
        dds = [r.max_drawdown_pct for r in results]
        
        returns_sorted = sorted(returns)
        dds_sorted = sorted(dds)
        n = len(results)
        
        # Return stats
        median_return = returns_sorted[n // 2]
        mean_return = sum(returns) / len(returns)
        return_5th = returns_sorted[n * 5 // 100]
        return_95th = returns_sorted[n * 95 // 100]
        return_std = self._std(returns)
        
        # DD stats
        median_dd = dds_sorted[n // 2]
        max_dd = max(dds)
        dd_95th = dds_sorted[n * 95 // 100]
        dd_99th = dds_sorted[n * 99 // 100]
        
        # Probabilities
        profitable = sum(1 for r in results if r.total_return > 0)
        prob_profit = profitable / n
        prob_dd_20 = sum(1 for dd in dds if dd > 20) / n
        prob_dd_30 = sum(1 for dd in dds if dd > 30) / n
        
        # Confidence intervals
        ci_return = (float(returns_sorted[n * 5 // 100]), float(returns_sorted[n * 95 // 100]))
        ci_dd = (float(dds_sorted[n * 5 // 100]), float(dds_sorted[n * 95 // 100]))
        
        return MCAnalysis(
            simulations=n,
            median_return=median_return,
            mean_return=mean_return,
            return_5th=return_5th,
            return_95th=return_95th,
            return_std=return_std,
            median_dd=median_dd,
            max_dd=max_dd,
            dd_95th=dd_95th,
            dd_99th=dd_99th,
            prob_profit=prob_profit,
            prob_dd_over_20=prob_dd_20,
            prob_dd_over_30=prob_dd_30,
            ci_return=ci_return,
            ci_dd=ci_dd,
            worst_return=min(returns),
            best_return=max(returns),
            worst_dd=max(dds)
        )
    
    def _std(self, values: list[Decimal]) -> Decimal:
        """Calculate standard deviation."""
        if len(values) < 2:
            return Decimal("0")
        m = sum(values) / len(values)
        variance = sum((v - m) ** 2 for v in values) / (len(values) - 1)
        return Decimal(str(math.sqrt(float(variance))))


def load_trades(csv_path: Path) -> list[Trade]:
    """Load trades from CSV."""
    trades = []
    with open(csv_path, "r") as f:
        reader = csv.DictReader(f)
        for row in reader:
            trades.append(Trade.from_csv_row(row))
    return trades


def print_analysis(analysis: MCAnalysis, original_trades: list[Trade] = None):
    """Print formatted analysis."""
    print(f"\n{'═'*60}")
    print(f"  MONTE CARLO ANALYSIS")
    print(f"  {analysis.simulations:,} Simulations")
    print(f"{'═'*60}")
    
    if original_trades:
        wins = sum(1 for t in original_trades if t.pnl > 0)
        total_pnl = sum(t.pnl for t in original_trades)
        print(f"\n  Original Backtest:")
        print(f"    Trades: {len(original_trades)}")
        print(f"    Win Rate: {wins/len(original_trades)*100:.1f}%")
        print(f"    Total P&L: ${total_pnl:,.2f}")
    
    print(f"\n{'─'*60}")
    print(f"  RETURN DISTRIBUTION")
    print(f"{'─'*60}")
    print(f"  {'Median:':<20} {analysis.median_return:+,.2f}%")
    print(f"  {'Mean:':<20} {analysis.mean_return:+,.2f}%")
    print(f"  {'90% CI:':<20} [{analysis.ci_return[0]:+.1f}% to {analysis.ci_return[1]:+.1f}%]")
    print(f"  {'Worst Case:':<20} {analysis.worst_return:+,.2f}%")
    print(f"  {'Best Case:':<20} {analysis.best_return:+,.2f}%")
    
    print(f"\n{'─'*60}")
    print(f"  DRAWDOWN DISTRIBUTION")
    print(f"{'─'*60}")
    print(f"  {'Median DD:':<20} {analysis.median_dd:.2f}%")
    print(f"  {'95th Percentile:':<20} {analysis.dd_95th:.2f}%")
    print(f"  {'99th Percentile:':<20} {analysis.dd_99th:.2f}%")
    print(f"  {'Maximum DD:':<20} {analysis.worst_dd:.2f}%")
    
    print(f"\n{'─'*60}")
    print(f"  PROBABILITIES")
    print(f"{'─'*60}")
    print(f"  {'Profitable:':<20} {analysis.prob_profit*100:.1f}%")
    print(f"  {'DD > 20%:':<20} {analysis.prob_dd_over_20*100:.1f}%")
    print(f"  {'DD > 30%:':<20} {analysis.prob_dd_over_30*100:.1f}%")
    
    # Interpretation
    print(f"\n{'─'*60}")
    print(f"  INTERPRETATION")
    print(f"{'─'*60}")
    
    if analysis.prob_profit > 0.9:
        print("  ✅ High probability of profitability")
    elif analysis.prob_profit > 0.7:
        print("  ⚠️  Moderate probability of profitability")
    else:
        print("  ❌ Low probability — results likely noise")
    
    if analysis.prob_dd_over_20 < 0.1:
        print("  ✅ Low probability of severe drawdown")
    elif analysis.prob_dd_over_20 < 0.3:
        print("  ⚠️  Moderate drawdown risk")
    else:
        print("  ❌ High probability (>30%) of 20%+ drawdown")
    
    print(f"{'═'*60}\n")


def main():
    parser = argparse.ArgumentParser(description="Monte Carlo Backtest Validator")
    parser.add_argument("csv", type=Path, help="Trades CSV (pnl column)")
    parser.add_argument("--simulations", "-n", type=int, default=10000, help="Sim count")
    parser.add_argument("--equity", "-e", type=float, default=10000, help="Start equity")
    parser.add_argument("--method", "-m", choices=["shuffle", "bootstrap"], 
                       default="shuffle", help="MC method")
    parser.add_argument("--seed", type=int, help="Random seed (reproducible)")
    parser.add_argument("--output", "-o", type=Path, help="JSON output")
    
    args = parser.parse_args()
    
    # Load and validate
    trades = load_trades(args.csv)
    if len(trades) < 10:
        print("Need at least 10 trades for meaningful MC analysis")
        return
    
    print(f"Loaded {len(trades)} trades")
    print(f"Running {args.simulations:,} simulations...")
    
    # Run MC
    validator = MCValidator(seed=args.seed)
    analysis = validator.run(
        trades,
        start_equity=Decimal(str(args.equity)),
        simulations=args.simulations,
        method=args.method
    )
    
    # Display
    print_analysis(analysis, trades)
    
    # Output
    if args.output:
        args.output.write_text(json.dumps(analysis.to_dict(), indent=2))
        print(f"💾 Saved to {args.output}")
```

## The Problem MC Solves

| Issue | MC Solution |
|-------|-------------|
| Backtest overfitting | Randomizes trade order |
| Lucky sequencing | Tests all possible paths |
| False confidence | Probability distributions instead of single line |
| Underestimated drawdown | Worst-case scenarios exposed |

## Methods

| Method | Use Case |
|--------|----------|
| **Shuffle** | Test if trade sequence matters |
| **Bootstrap** | Test with similar frequency distribution |

## Key Metrics

| Metric | Good | Bad |
|--------|------|-----|
| Prob Profit | > 80% | < 60% |
| 95th DD | < 20% | > 30% |
| 90% CI | Narrow positive | Includes heavy losses |

## When to Trust Your Backtest

✅ **Trust it** when:
- Prob profit > 80%
- 95% of DDs < 20%
- 90% CI stays positive

❌ **Don't trust** when:
- Prob profit < 60%
- Worst case DD > 50%
- Bootstrap results worse than shuffle
