# State Machine Pattern
*Ghost Learning | 2026-02-22*

Model trading entities with explicit states and transitions. Prevents invalid operations, clarifies logic.

## The Problem

```python
# Without state machine - bug prone
def place_order(order):
    if order.status == "filled":
        raise Error("Already filled")  # Easy to miss
    if order.status == "cancelled":
        raise Error("Already cancelled")
    # ... more checks
    order.status = "submitted"

# What transitions are valid? Unclear. Easy to skip checks.
```

## The Solution: Explicit State Machine

```python
"""
State Machine Pattern
Explicit states and transitions for trading systems.
"""

from dataclasses import dataclass, field
from typing import Callable, Dict, List, Optional, Set, Any
from enum import Enum, auto
from functools import wraps


class State(Enum):
    """Trading bot states."""
    IDLE = auto()
    ANALYZING = auto()
    SIGNAL_READY = auto()
    ORDER_PENDING = auto()
    POSITION_OPEN = auto()
    RISK_CHECK = auto()
    STOPPED = auto()
    ERROR = auto()


@dataclass
class Transition:
    """A state transition."""
    from_state: State
    to_state: State
    event: str
    guard: Optional[Callable[[], bool]] = None
    action: Optional[Callable[[], None]] = None


class StateMachine:
    """
    Generic state machine for trading systems.
    
    Usage:
        sm = StateMachine(State.IDLE)
        sm.add_transition(State.IDLE, State.ANALYZING, "start")
        sm.transition("start")  # IDLE -> ANALYZING
    """
    
    def __init__(self, initial_state: State):
        self.state = initial_state
        self.transitions: Dict[State, Dict[str, Transition]] = {}
        self.history: List[State] = [initial_state]
        self.context: Dict[str, Any] = {}
        
        # Callbacks
        self._on_enter: Dict[State, Callable] = {}
        self._on_exit: Dict[State, Callable] = {}
        self._on_transition: List[Callable] = []
    
    def add_transition(
        self,
        from_state: State,
        to_state: State,
        event: str,
        guard: Callable[[], bool] = None,
        action: Callable[[], None] = None
    ) -> "StateMachine":
        """Add a valid transition."""
        if from_state not in self.transitions:
            self.transitions[from_state] = {}
        
        self.transitions[from_state][event] = Transition(
            from_state=from_state,
            to_state=to_state,
            event=event,
            guard=guard,
            action=action
        )
        return self
    
    def on_enter(self, state: State, callback: Callable) -> "StateMachine":
        """Register callback for entering a state."""
        self._on_enter[state] = callback
        return self
    
    def on_exit(self, state: State, callback: Callable) -> "StateMachine":
        """Register callback for exiting a state."""
        self._on_exit[state] = callback
        return self
    
    def on_transition(self, callback: Callable) -> "StateMachine":
        """Register callback for any transition."""
        self._on_transition.append(callback)
        return self
    
    def can_transition(self, event: str) -> bool:
        """Check if transition is valid."""
        if self.state not in self.transitions:
            return False
        if event not in self.transitions[self.state]:
            return False
        
        transition = self.transitions[self.state][event]
        if transition.guard and not transition.guard():
            return False
        
        return True
    
    def transition(self, event: str) -> bool:
        """
        Attempt state transition.
        
        Returns True if successful.
        """
        if not self.can_transition(event):
            return False
        
        transition = self.transitions[self.state][event]
        old_state = self.state
        
        # Exit callback
        if old_state in self._on_exit:
            self._on_exit[old_state]()
        
        # Run action
        if transition.action:
            transition.action()
        
        # Change state
        self.state = transition.to_state
        self.history.append(self.state)
        
        # Enter callback
        if self.state in self._on_enter:
            self._on_enter[self.state]()
        
        # Transition callback
        for callback in self._on_transition:
            callback(old_state, self.state, event)
        
        return True
    
    def force_state(self, state: State) -> None:
        """Force state (for error recovery)."""
        old_state = self.state
        self.state = state
        self.history.append(state)
    
    def valid_events(self) -> Set[str]:
        """Get valid events from current state."""
        if self.state not in self.transitions:
            return set()
        return set(self.transitions[self.state].keys())
    
    def is_in(self, *states: State) -> bool:
        """Check if in any of the given states."""
        return self.state in states
    
    def reset(self, state: State = None) -> None:
        """Reset to initial or specified state."""
        self.state = state or self.history[0]
        self.history = [self.state]
        self.context.clear()


# === Trading Bot State Machine ===

class TradingBot:
    """
    Trading bot with state machine.
    """
    
    def __init__(self):
        self.sm = StateMachine(State.IDLE)
        self.order_id: Optional[str] = None
        self.position: Optional[dict] = None
        self.error_count = 0
        
        self._setup_transitions()
        self._setup_callbacks()
    
    def _setup_transitions(self):
        """Define valid state transitions."""
        (self.sm
            # IDLE -> ANALYZING
            .add_transition(State.IDLE, State.ANALYZING, "start")
            
            # ANALYZING -> SIGNAL_READY or IDLE
            .add_transition(State.ANALYZING, State.SIGNAL_READY, "signal_found")
            .add_transition(State.ANALYZING, State.IDLE, "no_signal")
            
            # SIGNAL_READY -> ORDER_PENDING or IDLE
            .add_transition(
                State.SIGNAL_READY, State.ORDER_PENDING, "place_order",
                guard=self._can_place_order,
                action=self._place_order
            )
            .add_transition(State.SIGNAL_READY, State.IDLE, "skip_signal")
            
            # ORDER_PENDING -> POSITION_OPEN or ERROR
            .add_transition(
                State.ORDER_PENDING, State.POSITION_OPEN, "order_filled",
                action=self._on_fill
            )
            .add_transition(
                State.ORDER_PENDING, State.IDLE, "order_cancelled",
                action=self._on_cancel
            )
            .add_transition(State.ORDER_PENDING, State.ERROR, "order_failed")
            
            # POSITION_OPEN -> IDLE
            .add_transition(
                State.POSITION_OPEN, State.IDLE, "close_position",
                action=self._close_position
            )
            .add_transition(State.POSITION_OPEN, State.RISK_CHECK, "risk_alert")
            
            # RISK_CHECK -> POSITION_OPEN or IDLE
            .add_transition(State.RISK_CHECK, State.POSITION_OPEN, "risk_ok")
            .add_transition(State.RISK_CHECK, State.IDLE, "risk_exit", action=self._close_position)
            
            # ERROR handling
            .add_transition(State.ERROR, State.IDLE, "reset")
            .add_transition(State.ERROR, State.STOPPED, "stop")
            
            # Any state -> STOPPED
            .add_transition(State.IDLE, State.STOPPED, "stop")
            .add_transition(State.ANALYZING, State.STOPPED, "stop")
        )
    
    def _setup_callbacks(self):
        """Set up state callbacks."""
        (self.sm
            .on_enter(State.ANALYZING, lambda: print("📊 Analyzing market..."))
            .on_enter(State.SIGNAL_READY, lambda: print("✅ Signal ready!"))
            .on_enter(State.ORDER_PENDING, lambda: print("📤 Order placed, waiting..."))
            .on_enter(State.POSITION_OPEN, lambda: print("📈 Position open"))
            .on_enter(State.ERROR, lambda: print("❌ Error state"))
            .on_enter(State.STOPPED, lambda: print("🛑 Bot stopped"))
        )
    
    def _can_place_order(self) -> bool:
        """Guard: Check if order can be placed."""
        # Add risk checks here
        return self.error_count < 3
    
    def _place_order(self):
        """Action: Place order."""
        self.order_id = f"ORD-{id(self) % 10000}"
        print(f"   Placing order {self.order_id}")
    
    def _on_fill(self):
        """Action: Order filled."""
        self.position = {"order_id": self.order_id, "size": 1.0}
        print(f"   Order {self.order_id} filled")
    
    def _on_cancel(self):
        """Action: Order cancelled."""
        print(f"   Order {self.order_id} cancelled")
        self.order_id = None
    
    def _close_position(self):
        """Action: Close position."""
        if self.position:
            print(f"   Closing position from {self.position['order_id']}")
            self.position = None
    
    # Public methods delegate to state machine
    
    def start(self) -> bool:
        return self.sm.transition("start")
    
    def signal_found(self) -> bool:
        return self.sm.transition("signal_found")
    
    def place_order(self) -> bool:
        return self.sm.transition("place_order")
    
    def order_filled(self) -> bool:
        return self.sm.transition("order_filled")
    
    def close_position(self) -> bool:
        return self.sm.transition("close_position")
    
    def stop(self) -> bool:
        return self.sm.transition("stop")
    
    @property
    def state(self) -> State:
        return self.sm.state


# === Decorator for State Guards ===

def require_state(*allowed_states: State):
    """Decorator to require specific state."""
    def decorator(func):
        @wraps(func)
        def wrapper(self, *args, **kwargs):
            if not self.sm.is_in(*allowed_states):
                raise ValueError(
                    f"Cannot call {func.__name__} in state {self.sm.state.name}. "
                    f"Required: {[s.name for s in allowed_states]}"
                )
            return func(self, *args, **kwargs)
        return wrapper
    return decorator


# === Usage ===

if __name__ == "__main__":
    print("=== TRADING BOT STATE MACHINE ===\n")
    
    bot = TradingBot()
    print(f"Initial state: {bot.state.name}")
    print(f"Valid events: {bot.sm.valid_events()}\n")
    
    # Full trading cycle
    print("--- Full Trading Cycle ---")
    bot.start()          # IDLE -> ANALYZING
    bot.signal_found()   # ANALYZING -> SIGNAL_READY
    bot.place_order()    # SIGNAL_READY -> ORDER_PENDING
    bot.order_filled()   # ORDER_PENDING -> POSITION_OPEN
    bot.close_position() # POSITION_OPEN -> IDLE
    
    print(f"\nFinal state: {bot.state.name}")
    print(f"History: {[s.name for s in bot.sm.history]}")
    
    print("\n--- Invalid Transitions ---")
    bot = TradingBot()
    print(f"State: {bot.state.name}")
    
    # Try to place order from IDLE (should fail)
    success = bot.place_order()
    print(f"place_order() from IDLE: {success} (expected False)")
    
    print("\n--- Guard Check ---")
    bot = TradingBot()
    bot.start()
    bot.signal_found()
    bot.error_count = 5  # Exceed limit
    success = bot.place_order()
    print(f"place_order() with error_count=5: {success} (expected False, guard blocked)")
    
    print("\n--- ERROR handling ---")
    bot = TradingBot()
    bot.start()
    bot.signal_found()
    bot.place_order()
    bot.sm.transition("order_failed")  # Force error
    print(f"State after failure: {bot.state.name}")
    bot.sm.transition("reset")
    print(f"State after reset: {bot.state.name}")
    
    print("\n=== STATE MACHINE REFERENCE ===")
    print("""
```

## Key Concepts

| Concept | Purpose |
|---------|---------|
| **State** | Current condition (exclusive) |
| **Transition** | Valid state change triggered by event |
| **Guard** | Condition that must be true to transition |
| **Action** | Code to run during transition |
| **Callback** | Code to run on enter/exit state |

## Quick Reference

```python
sm = StateMachine(State.IDLE)

# Add transitions
sm.add_transition(State.IDLE, State.RUNNING, "start")
sm.add_transition(State.RUNNING, State.IDLE, "stop")

# With guard and action
sm.add_transition(State.ORDERING, State.FILLED, "fill",
    guard=lambda: can_fill(),
    action=lambda: process_fill())

# Callbacks
sm.on_enter(State.RUNNING, lambda: print("Running"))
sm.on_exit(State.RUNNING, lambda: cleanup())

# Transition
if sm.can_transition("start"):
    sm.transition("start")

# Query
sm.state                    # Current state
sm.valid_events()           # Valid events
sm.is_in(State.IDLE)        # Boolean check
sm.history                  # State history
```

## When to Use

| Use Case | Example |
|----------|---------|
| Order lifecycle | pending → filled/cancelled |
| Bot lifecycle | idle → running → stopped |
| Connection status | disconnected → connecting → connected |
| Position management | flat → opening → open → closing |
| Risk states | normal → warning → critical |
