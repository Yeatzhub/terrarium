```python
"""
Trade Journal
Log and analyze trading performance.
"""

from dataclasses import dataclass, field
from datetime import datetime
from decimal import Decimal
from typing import Optional
from enum import Enum
import json
import csv
from pathlib import Path


class Side(Enum):
    LONG = "long"
    SHORT = "short"


class Status(Enum):
    OPEN = "open"
    CLOSED = "closed"
    CANCELLED = "cancelled"


@dataclass
class Trade:
    """Single trade record."""
    # Identity
    trade_id: str
    symbol: str
    side: Side
    status: Status = Status.OPEN
    
    # Entry
    entry_time: datetime = None
    entry_price: Decimal = Decimal("0")
    position_size: Decimal = Decimal("0")
    stop_loss: Optional[Decimal] = None
    take_profit: Optional[Decimal] = None
    
    # Exit
    exit_time: Optional[datetime] = None
    exit_price: Optional[Decimal] = None
    
    # Risk
    account_balance: Decimal = Decimal("0")
    risk_per_trade: Decimal = Decimal("0")
    
    # Metadata
    strategy: str = ""
    setup: str = ""  # e.g., "breakout", "reversal", "range"
    tags: list[str] = field(default_factory=list)
    notes: str = ""
    
    # Calculated (set after exit)
    pnl: Optional[Decimal] = None
    pnl_pct: Optional[Decimal] = None
    r_multiple: Optional[Decimal] = None
    
    def __post_init__(self):
        if self.entry_time is None:
            self.entry_time = datetime.utcnow()
    
    @property
    def risk_amount(self) -> Decimal:
        """Dollar risk if stopped."""
        if self.stop_loss and self.position_size:
            return abs(self.entry_price - self.stop_loss) * self.position_size
        return Decimal("0")
    
    @property
    def position_value(self) -> Decimal:
        """Position value at entry."""
        return self.entry_price * self.position_size
    
    @property
    def hold_time_hours(self) -> Optional[float]:
        """Hours held."""
        if self.exit_time and self.entry_time:
            return (self.exit_time - self.entry_time).total_seconds() / 3600
        return None
    
    def close(
        self,
        exit_price: Decimal,
        exit_time: Optional[datetime] = None
    ) -> "Trade":
        """Close the trade and calculate P&L."""
        self.exit_price = exit_price
        self.exit_time = exit_time or datetime.utcnow()
        self.status = Status.CLOSED
        
        # Calculate P&L
        if self.side == Side.LONG:
            self.pnl = (exit_price - self.entry_price) * self.position_size
        else:
            self.pnl = (self.entry_price - exit_price) * self.position_size
        
        # P&L percentage
        self.pnl_pct = self.pnl / self.position_value if self.position_value else Decimal("0")
        
        # R-multiple
        if self.risk_amount > 0:
            self.r_multiple = self.pnl / self.risk_amount
        
        return self
    
    def to_dict(self) -> dict:
        """Serialize to dictionary."""
        return {
            "trade_id": self.trade_id,
            "symbol": self.symbol,
            "side": self.side.value,
            "status": self.status.value,
            "entry_time": self.entry_time.isoformat() if self.entry_time else None,
            "entry_price": float(self.entry_price),
            "position_size": float(self.position_size),
            "stop_loss": float(self.stop_loss) if self.stop_loss else None,
            "take_profit": float(self.take_profit) if self.take_profit else None,
            "exit_time": self.exit_time.isoformat() if self.exit_time else None,
            "exit_price": float(self.exit_price) if self.exit_price else None,
            "account_balance": float(self.account_balance),
            "strategy": self.strategy,
            "setup": self.setup,
            "tags": self.tags,
            "notes": self.notes,
            "pnl": float(self.pnl) if self.pnl else None,
            "pnl_pct": float(self.pnl_pct) if self.pnl_pct else None,
            "r_multiple": float(self.r_multiple) if self.r_multiple else None,
        }


@dataclass
class JournalStats:
    """Performance statistics."""
    total_trades: int
    winners: int
    losers: int
    win_rate: Decimal
    total_pnl: Decimal
    avg_win: Decimal
    avg_loss: Decimal
    profit_factor: Decimal
    expectancy: Decimal
    avg_r_multiple: Decimal
    best_trade: Optional[Decimal]
    worst_trade: Optional[Decimal]
    avg_hold_hours: float
    
    def __str__(self) -> str:
        lines = [
            "=== JOURNAL STATS ===",
            f"Trades:       {self.total_trades} ({self.winners}W/{self.losers}L)",
            f"Win Rate:     {self.win_rate:.1%}",
            f"Total P&L:    ${self.total_pnl:,.2f}",
            f"",
            f"Avg Win:      ${self.avg_win:,.2f}",
            f"Avg Loss:     ${self.avg_loss:,.2f}",
            f"Profit Factor: {self.profit_factor:.2f}",
            f"Expectancy:   ${self.expectancy:,.2f}",
            f"",
            f"Avg R:        {self.avg_r_multiple:.2f}",
            f"Best Trade:   ${self.best_trade:,.2f}" if self.best_trade else "Best Trade:   N/A",
            f"Worst Trade:  ${self.worst_trade:,.2f}" if self.worst_trade else "Worst Trade:  N/A",
            f"Avg Hold:     {self.avg_hold_hours:.1f}h",
        ]
        return "\n".join(lines)


class TradeJournal:
    """
    Trade journal with analytics.
    """
    
    def __init__(self):
        self.trades: dict[str, Trade] = {}
        self._counter = 0
    
    def _next_id(self) -> str:
        self._counter += 1
        return f"T{datetime.utcnow().strftime('%Y%m%d')}-{self._counter:04d}"
    
    def open_trade(
        self,
        symbol: str,
        side: Side,
        entry_price: Decimal,
        position_size: Decimal,
        stop_loss: Optional[Decimal] = None,
        take_profit: Optional[Decimal] = None,
        strategy: str = "",
        setup: str = "",
        tags: list[str] = None,
        notes: str = "",
        account_balance: Decimal = Decimal("0"),
    ) -> Trade:
        """Open a new trade."""
        trade = Trade(
            trade_id=self._next_id(),
            symbol=symbol,
            side=side,
            entry_price=entry_price,
            position_size=position_size,
            stop_loss=stop_loss,
            take_profit=take_profit,
            strategy=strategy,
            setup=setup,
            tags=tags or [],
            notes=notes,
            account_balance=account_balance,
        )
        self.trades[trade.trade_id] = trade
        return trade
    
    def close_trade(
        self,
        trade_id: str,
        exit_price: Decimal,
        exit_time: Optional[datetime] = None,
        notes: str = ""
    ) -> Trade:
        """Close an open trade."""
        if trade_id not in self.trades:
            raise ValueError(f"Trade {trade_id} not found")
        
        trade = self.trades[trade_id]
        if notes:
            trade.notes = f"{trade.notes}\n{notes}".strip()
        return trade.close(exit_price, exit_time)
    
    def get_open_trades(self) -> list[Trade]:
        """Get all open trades."""
        return [t for t in self.trades.values() if t.status == Status.OPEN]
    
    def get_closed_trades(self) -> list[Trade]:
        """Get all closed trades."""
        return [t for t in self.trades.values() if t.status == Status.CLOSED]
    
    def calculate_stats(
        self,
        symbol: Optional[str] = None,
        strategy: Optional[str] = None,
        tag: Optional[str] = None,
    ) -> JournalStats:
        """Calculate performance statistics with optional filters."""
        closed = self.get_closed_trades()
        
        # Apply filters
        if symbol:
            closed = [t for t in closed if t.symbol == symbol]
        if strategy:
            closed = [t for t in closed if t.strategy == strategy]
        if tag:
            closed = [t for t in closed if tag in t.tags]
        
        if not closed:
            return JournalStats(
                total_trades=0, winners=0, losers=0,
                win_rate=Decimal("0"), total_pnl=Decimal("0"),
                avg_win=Decimal("0"), avg_loss=Decimal("0"),
                profit_factor=Decimal("0"), expectancy=Decimal("0"),
                avg_r_multiple=Decimal("0"), best_trade=None, worst_trade=None,
                avg_hold_hours=0
            )
        
        wins = [t.pnl for t in closed if t.pnl and t.pnl > 0]
        losses = [t.pnl for t in closed if t.pnl and t.pnl < 0]
        
        total_pnl = sum(t.pnl for t in closed if t.pnl) or Decimal("0")
        total_wins = sum(wins) or Decimal("0")
        total_losses = sum(losses) or Decimal("0")
        
        profit_factor = abs(total_wins / total_losses) if total_losses else Decimal("999")
        
        r_multiples = [t.r_multiple for t in closed if t.r_multiple]
        avg_r = sum(r_multiples) / len(r_multiples) if r_multiples else Decimal("0")
        
        hold_times = [t.hold_time_hours for t in closed if t.hold_time_hours]
        avg_hold = sum(hold_times) / len(hold_times) if hold_times else 0
        
        return JournalStats(
            total_trades=len(closed),
            winners=len(wins),
            losers=len(losses),
            win_rate=Decimal(len(wins)) / Decimal(len(closed)),
            total_pnl=total_pnl,
            avg_win=total_wins / len(wins) if wins else Decimal("0"),
            avg_loss=total_losses / len(losses) if losses else Decimal("0"),
            profit_factor=profit_factor,
            expectancy=total_pnl / len(closed),
            avg_r_multiple=avg_r,
            best_trade=max(wins) if wins else None,
            worst_trade=min(losses) if losses else None,
            avg_hold_hours=avg_hold
        )
    
    def export_csv(self, filepath: str) -> None:
        """Export to CSV."""
        closed = self.get_closed_trades()
        with open(filepath, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=list(Trade.__dataclass_fields__.keys()))
            writer.writeheader()
            for trade in closed:
                writer.writeheader()
        with open(filepath, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=[
                "trade_id", "symbol", "side", "entry_time", "entry_price",
                "position_size", "stop_loss", "take_profit", "exit_time",
                "exit_price", "pnl", "pnl_pct", "r_multiple", "strategy",
                "setup", "tags", "notes"
            ])
            writer.writeheader()
            for trade in closed:
                row = trade.to_dict()
                row["tags"] = ",".join(trade.tags)
                writer.writerow(row)
```

## Key Metrics

| Metric | Formula | Good |
|--------|---------|------|
| Win Rate | Wins / Total | >45% |
| Profit Factor | Gross Win / Gross Loss | >1.5 |
| Expectancy | Avg $ per trade | >0 |
| Avg R | Mean R-multiple | >1.0 |
