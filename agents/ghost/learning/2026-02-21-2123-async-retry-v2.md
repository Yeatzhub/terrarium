"""
Async Retry Pattern v2 — Enhanced with Metrics and Adaptive Backoff

import asyncio
import random
import functools
import time
from dataclasses import dataclass, field, asdict
from typing import TypeVar, Callable, Optional, Tuple, Type, Dict, Any
from datetime import datetime, timedelta
from collections import defaultdict
from statistics import mean


T = TypeVar("T")


@dataclass
class RetryMetrics:
    """Metrics for a single retry operation."""
    function_name: str
    attempts: int
    total_duration_ms: float
    final_success: bool
    exception_type: Optional[str] = None
    circuit_open: bool = False
    timestamp: datetime = field(default_factory=datetime.utcnow)


@dataclass
class FunctionStats:
    """Aggregated stats for a decorated function."""
    function_name: str
    total_calls: int = 0
    successes: int = 0
    failures: int = 0
    circuit_opens: int = 0
    
    # Timing
    attempt_counts: list = field(default_factory=list)
    durations_ms: list = field(default_factory=list)
    
    @property
    def success_rate(self) -> float:
        return self.successes / self.total_calls if self.total_calls > 0 else 0
    
    @property
    def avg_attempts(self) -> float:
        return mean(self.attempt_counts) if self.attempt_counts else 0
    
    @property
    def avg_duration_ms(self) -> float:
        return mean(self.durations_ms) if self.durations_ms else 0
    
    @property
    def error_rate(self) -> float:
        return self.failures / self.total_calls if self.total_calls > 0 else 0
    
    def to_dict(self) -
        return {
            "function": self.function_name,
            "total_calls": self.total_calls,
            "success_rate": f"{self.success_rate*100:.1f}%",
            "error_rate": f"{self.error_rate*100:.1f}%",
            "avg_attempts": f"{self.avg_attempts:.1f}",
            "avg_duration_ms": f"{self.avg_duration_ms:.0f}",
            "circuit_opens": self.circuit_opens
        }


@dataclass
class RetryConfig:
    """Enhanced retry configuration."""
    max_attempts: int = 3
    base_delay: float = 1.0
    max_delay: float = 60.0
    exponential_base: float = 2.0
    jitter: bool = True
    jitter_max: float = 1.0
    exceptions: Tuple[Type[Exception], ...] = (Exception,)
    
    # Adaptive backoff
    adaptive: bool = True                    # Increase delay based on error rate
    adaptive_threshold: float = 0.5          # Error rate > 50% triggers adapt
    adaptive_multiplier: float = 2.0         # Multiply base delay by this
    
    # Callbacks
    on_retry: Optional[Callable[[Exception, int], None]] = None
    on_giveup: Optional[Callable[[Exception], None]] = None
    on_success: Optional[Callable[[int, float], None]] = None  # (attempts, duration_ms)


class GlobalMetricsRegistry:
    """Global registry for all retry metrics."""
    _instance: Optional["GlobalMetricsRegistry"] = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._reset()
        return cls._instance
    
    def _reset(self):
        self.stats: Dict[str, FunctionStats] = defaultdict(
            lambda: FunctionStats(function_name="unknown")
        )
        self.recent_metrics: list[RetryMetrics] = []
        self.max_recent = 1000
    
    def record(self, metric: RetryMetrics):
        """Record a metric."""
        func_name = metric.function_name
        
        # Initialize if first time
        if func_name not in self.stats or self.stats[func_name].function_name == "unknown":
            self.stats[func_name] = FunctionStats(function_name=func_name)
        
        stats = self.stats[func_name]
        stats.total_calls += 1
        
        if metric.final_success:
            stats.successes += 1
        else:
            stats.failures += 1
        
        if metric.circuit_open:
            stats.circuit_opens += 1
        
        stats.attempt_counts.append(metric.attempts)
        stats.durations_ms.append(metric.total_duration_ms)
        
        # Keep recent metrics bounded
        self.recent_metrics.append(metric)
        if len(self.recent_metrics) > self.max_recent:
            self.recent_metrics.pop(0)
    
    def get_stats(self, function_name: Optional[str] = None) -> Dict[str, Any]:
        """Get stats for a function or all functions."""
        if function_name:
            return self.stats.get(function_name, FunctionStats(function_name)).to_dict()
        return {name: stats.to_dict() for name, stats in self.stats.items()}
    
    def get_error_rate(self, function_name: str, window: int = 100) -> float:
        """Get error rate for recent calls."""
        recent = [m for m in self.recent_metrics[-window:] 
                 if m.function_name == function_name]
        if not recent:
            return 0.0
        failures = sum(1 for m in recent if not m.final_success)
        return failures / len(recent)
    
    def print_summary(self):
        """Print stats summary."""
        print(f"\n{'═'*60}")
        print(f"  RETRY METRICS SUMMARY")
        print(f"{'═'*60}")
        for name, stats in self.stats.items():
            if stats.total_calls > 0:
                print(f"  {name:<30} {stats.success_rate*100:5.1f}% success ({stats.total_calls} calls)")
        print(f"{'═'*60}\n")


metrics_registry = GlobalMetricsRegistry()


class CircuitOpenError(Exception):
    """Circuit breaker is open."""
    pass


@dataclass
class CircuitBreaker:
    """Enhanced circuit breaker with state history."""
    failure_threshold: int = 5
    recovery_timeout: float = 60.0
    name: str = "default"
    
    _failures: int = field(default=0, repr=False)
    _last_failure_time: Optional[datetime] = field(default=None, repr=False)
    _state: str = field(default="CLOSED", repr=False)
    _state_changes: list = field(default_factory=list, repr=False)
    _successes: int = field(default=0, repr=False)
    
    def record_success(self) -> None:
        """Reset on success."""
        if self._failures > 0:
            self._state_changes.append((datetime.utcnow(), "CLOSED"))
        self._failures = 0
        self._state = "CLOSED"
        self._last_failure_time = None
        self._successes += 1
    
    def record_failure(self) -> None:
        """Increment failures."""
        self._failures += 1
        self._last_failure_time = datetime.utcnow()
        
        if self._failures >= self.failure_threshold and self._state != "OPEN":
            self._state = "OPEN"
            self._state_changes.append((datetime.utcnow(), "OPEN"))
    
    def can_execute(self) -> bool:
        """Check if execution should proceed."""
        if self._state == "CLOSED":
            return True
        
        if self._state == "OPEN":
            if self._last_failure_time:
                elapsed = (datetime.utcnow() - self._last_failure_time).total_seconds()
                if elapsed >= self.recovery_timeout:
                    self._state = "HALF_OPEN"
                    self._state_changes.append((datetime.utcnow(), "HALF_OPEN"))
                    return True
            return False
        
        return True  # HALF_OPEN
    
    @property
    def state(self) -> str:
        return self._state
    
    def get_stats(self) -> dict:
        return {
            "name": self.name,
            "state": self._state,
            "failures": self._failures,
            "successes": self._successes,
            "state_changes": len(self._state_changes)
        }


def calculate_adaptive_delay(
    attempt: int, 
    config: RetryConfig,
    function_name: str,
    registry: GlobalMetricsRegistry
) -> float:
    """Calculate delay with adaptive backoff based on error rate."""
    # Base exponential delay
    base_delay = config.base_delay * (config.exponential_base ** (attempt - 1))
    delay = min(base_delay, config.max_delay)
    
    # Adaptive: increase delay if error rate is high
    if config.adaptive:
        error_rate = registry.get_error_rate(function_name, window=50)
        if error_rate > config.adaptive_threshold:
            delay *= config.adaptive_multiplier
            delay = min(delay, config.max_delay)
    
    if config.jitter:
        jitter = random.uniform(-config.jitter_max / 2, config.jitter_max / 2)
        delay = max(0, delay + jitter)
    
    return delay


def retry(
    max_attempts: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    exponential_base: float = 2.0,
    jitter: bool = True,
    jitter_max: float = 1.0,
    exceptions: Tuple[Type[Exception], ...] = (Exception,),
    circuit_breaker: Optional[CircuitBreaker] = None,
    adaptive: bool = True,
    adaptive_threshold: float = 0.5,
    adaptive_multiplier: float = 2.0,
    on_retry: Optional[Callable[[Exception, int], None]] = None,
    on_giveup: Optional[Callable[[Exception], None]] = None,
    on_success: Optional[Callable[[int, float], None]] = None,
    registry: GlobalMetricsRegistry = None
):
    """Enhanced retry decorator with metrics and adaptive backoff."""
    
    config = RetryConfig(
        max_attempts=max_attempts,
        base_delay=base_delay,
        max_delay=max_delay,
        exponential_base=exponential_base,
        jitter=jitter,
        jitter_max=jitter_max,
        exceptions=exceptions,
        adaptive=adaptive,
        adaptive_threshold=adaptive_threshold,
        adaptive_multiplier=adaptive_multiplier,
        on_retry=on_retry,
        on_giveup=on_giveup,
        on_success=on_success
    )
    
    reg = registry or metrics_registry
    
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        func_name = func.__name__
        
        @functools.wraps(func)
        async def wrapper(*args, **kwargs) -> T:
            start_time = time.perf_counter()
            last_exception: Optional[Exception] = None
            attempts = 0
            circuit_was_open = False
            
            # Check circuit breaker
            if circuit_breaker and not circuit_breaker.can_execute():
                circuit_was_open = True
                metric = RetryMetrics(
                    function_name=func_name,
                    attempts=0,
                    total_duration_ms=(time.perf_counter() - start_time) * 1000,
                    final_success=False,
                    exception_type="CircuitOpenError",
                    circuit_open=True
                )
                reg.record(metric)
                raise CircuitOpenError(f"Circuit breaker open for {func_name}")
            
            for attempt in range(1, config.max_attempts + 1):
                attempts = attempt
                try:
                    result = await func(*args, **kwargs)
                    
                    # Record success
                    if circuit_breaker:
                        circuit_breaker.record_success()
                    
                    duration_ms = (time.perf_counter() - start_time) * 1000
                    
                    # Callback
                    if config.on_success:
                        config.on_success(attempt, duration_ms)
                    
                    # Record metric
                    metric = RetryMetrics(
                        function_name=func_name,
                        attempts=attempt,
                        total_duration_ms=duration_ms,
                        final_success=True
                    )
                    reg.record(metric)
                    
                    return result
                    
                except config.exceptions as e:
                    last_exception = e
                    
                    if circuit_breaker:
                        circuit_breaker.record_failure()
                        if not circuit_breaker.can_execute():
                            break
                    
                    if attempt == config.max_attempts:
                        break
                    
                    if config.on_retry:
                        config.on_retry(e, attempt)
                    
                    # Adaptive delay
                    delay = calculate_adaptive_delay(attempt, config, func_name, reg)
                    await asyncio.sleep(delay)
            
            # All retries exhausted
            duration_ms = (time.perf_counter() - start_time) * 1000
            
            if config.on_giveup:
                config.on_giveup(last_exception)
            
            # Record failure metric
            metric = RetryMetrics(
                function_name=func_name,
                attempts=attempts,
                total_duration_ms=duration_ms,
                final_success=False,
                exception_type=type(last_exception).__name__ if last_exception else None,
                circuit_open=circuit_was_open
            )
            reg.record(metric)
            
            raise last_exception
        
        return wrapper
    return decorator

## Adaptive Backoff

When error rate > threshold (default 50%):
- Base delay multiplied by `adaptive_multiplier` (default 2x)
- Helps when downstream service is struggling
- Automatically backs off when errors spike
