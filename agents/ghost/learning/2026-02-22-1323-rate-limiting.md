# Rate Limiting - Token Bucket Pattern
*Ghost Learning | 2026-02-22*

Control request flow to APIs, services, or any rate-limited resource.

## The Problem

APIs have rate limits: 100 requests/minute, 10 orders/second, etc.
Exceed limits → errors, bans, lost opportunities.

## Token Bucket Algorithm

```
Bucket holds tokens (max: capacity)
Tokens refill at rate (tokens/second)
Each request costs 1 token
No tokens? Wait or reject.
```

## Implementation

```python
"""
Rate Limiter - Token Bucket Pattern
Smoothing request flow with configurable limits.
"""

import time
import threading
from dataclasses import dataclass, field
from decimal import Decimal
from typing import Optional, Callable
from enum import Enum
from contextlib import contextmanager


class LimitStrategy(Enum):
    WAIT = "wait"      # Block until token available
    REJECT = "reject"  # Return None immediately
    BURST = "burst"    # Allow temporary overrun


@dataclass
class RateLimitConfig:
    """Rate limit configuration."""
    requests_per_period: int
    period_seconds: float
    
    @property
    def rate_per_second(self) -> float:
        return self.requests_per_period / self.period_seconds


class TokenBucket:
    """
    Thread-safe token bucket rate limiter.
    
    Usage:
        limiter = TokenBucket(rate=10, capacity=20)
        with limiter:
            make_request()
    """
    
    def __init__(
        self,
        rate: float,          # Tokens added per second
        capacity: int,        # Max tokens in bucket
        strategy: LimitStrategy = LimitStrategy.WAIT,
    ):
        self.rate = rate
        self.capacity = capacity
        self.strategy = strategy
        
        self._tokens = float(capacity)
        self._last_update = time.monotonic()
        self._lock = threading.Lock()
        
        # Stats
        self._total_requests = 0
        self._blocked_requests = 0
        self._total_wait_time = 0.0
    
    def _refill(self) -> None:
        """Add tokens based on elapsed time."""
        now = time.monotonic()
        elapsed = now - self._last_update
        self._tokens = min(
            self.capacity,
            self._tokens + (elapsed * self.rate)
        )
        self._last_update = now
    
    def acquire(self, tokens: int = 1, timeout: Optional[float] = None) -> bool:
        """
        Acquire tokens.
        
        Returns True if acquired, False if rejected/timeout.
        """
        with self._lock:
            self._refill()
            self._total_requests += 1
            
            if self._tokens >= tokens:
                self._tokens -= tokens
                return True
            
            # Not enough tokens
            if self.strategy == LimitStrategy.REJECT:
                self._blocked_requests += 1
                return False
            
            if self.strategy == LimitStrategy.WAIT:
                # Calculate wait time
                needed = tokens - self._tokens
                wait_time = needed / self.rate
                
                if timeout and wait_time > timeout:
                    self._blocked_requests += 1
                    return False
                
                # Release lock and wait
                self._lock.release()
                try:
                    time.sleep(wait_time)
                    self._total_wait_time += wait_time
                finally:
                    self._lock.acquire()
                
                # Recalculate after wait
                self._refill()
                if self._tokens >= tokens:
                    self._tokens -= tokens
                    return True
                
                self._blocked_requests += 1
                return False
            
            # BURST strategy - allow with debt
            if self.strategy == LimitStrategy.BURST:
                self._tokens -= tokens
                return True
        
        return False
    
    def try_acquire(self, tokens: int = 1) -> bool:
        """Non-blocking attempt to acquire tokens."""
        return self.acquire(tokens, timeout=0)
    
    def wait_time(self, tokens: int = 1) -> float:
        """Time until tokens available."""
        with self._lock:
            self._refill()
            if self._tokens >= tokens:
                return 0.0
            needed = tokens - self._tokens
            return needed / self.rate
    
    @contextmanager
    def limit(self, tokens: int = 1, timeout: Optional[float] = None):
        """Context manager for rate-limited operations."""
        acquired = self.acquire(tokens, timeout)
        try:
            yield acquired
        finally:
            pass  # Tokens consumed
    
    def stats(self) -> dict:
        """Usage statistics."""
        with self._lock:
            return {
                "tokens_available": self._tokens,
                "capacity": self.capacity,
                "rate_per_second": self.rate,
                "total_requests": self._total_requests,
                "blocked_requests": self._blocked_requests,
                "total_wait_time": self._total_wait_time,
                "utilization": self._total_requests / max(1, self._total_requests + self._blocked_requests)
            }


class MultiRateLimiter:
    """
    Multiple rate limiters with priorities.
    
    Different limits for different operations.
    """
    
    def __init__(self):
        self.limiters: dict[str, TokenBucket] = {}
    
    def add_limit(
        self,
        name: str,
        rate: float,
        capacity: int,
        strategy: LimitStrategy = LimitStrategy.WAIT,
    ) -> "MultiRateLimiter":
        """Add a named rate limit."""
        self.limiters[name] = TokenBucket(rate, capacity, strategy)
        return self
    
    def acquire(self, *names: str) -> bool:
        """Acquire from multiple limiters atomically."""
        # Check all available first
        for name in names:
            if name not in self.limiters:
                raise ValueError(f"Unknown limit: {name}")
        
        # Try non-blocking first
        for name in names:
            if not self.limiters[name].try_acquire():
                # Rollback and wait
                for prev in names:
                    if prev == name:
                        break
                    # Can't rollback in token bucket, so we accept overage
                    pass
                
                # Now wait for each
                for n in names:
                    self.limiters[n].acquire()
                return True
        
        return True
    
    @contextmanager
    def limit(self, *names: str):
        """Context manager for multi-limiter."""
        self.acquire(*names)
        yield


# === Convenience Factory ===

def rate_limit(
    requests: int,
    per_seconds: float = 1.0,
    burst: int = None
) -> TokenBucket:
    """Create rate limiter from human-friendly params."""
    rate = requests / per_seconds
    capacity = burst or requests
    return TokenBucket(rate=rate, capacity=capacity)


# === Decorator ===

def rate_limited(
    requests: int,
    per_seconds: float = 1.0,
    key_func: Optional[Callable] = None
):
    """
    Decorator to rate limit a function.
    
    Usage:
        @rate_limited(10, per_seconds=1.0)
        def api_call():
            ...
    """
    limiter = rate_limit(requests, per_seconds)
    
    def decorator(func: Callable) -> Callable:
        def wrapper(*args, **kwargs):
            with limiter.limit():
                return func(*args, **kwargs)
        return wrapper
    return decorator


# === Usage ===

if __name__ == "__main__":
    print("=== BASIC RATE LIMITER ===")
    limiter = rate_limit(5, per_seconds=1.0)  # 5/sec
    
    start = time.time()
    for i in range(10):
        with limiter.limit() as acquired:
            elapsed = time.time() - start
            print(f"Request {i+1}: {elapsed:.2f}s - acquired={acquired}")
    
    print("\n=== STATS ===")
    stats = limiter.stats()
    print(f"Total requests: {stats['total_requests']}")
    print(f"Blocked: {stats['blocked_requests']}")
    print(f"Utilization: {stats['utilization']:.1%}")
    
    print("\n=== MULTIPLE LIMITS ===")
    multi = (MultiRateLimiter()
        .add_limit("orders", rate=2, capacity=5)    # 2 orders/sec
        .add_limit("api", rate=10, capacity=20))   # 10 API calls/sec
    
    # Acquire from both
    with multi.limit("orders", "api"):
        print("Made order with API check")
    
    print("\n=== DECORATOR ===")
    @rate_limited(3, per_seconds=1.0)
    def slow_api():
        return "result"
    
    start = time.time()
    for i in range(5):
        result = slow_api()
        elapsed = time.time() - start
        print(f"Call {i+1}: {elapsed:.2f}s")
    
    print("\n=== QUICK REFERENCE ===")
    # 10 requests per second, can burst to 20
    limiter = TokenBucket(rate=10, capacity=20)
    
    # Non-blocking check
    if limiter.try_acquire():
        print("Token acquired!")
    else:
        print("Rate limited, need to wait")
    
    # Check wait time
    print(f"Wait time: {limiter.wait_time():.2f}s")
    
    # Context manager
    with limiter.limit():
        make_request = "do_something"
        print(f"In rate-limited block: {make_request}")
```

## Common Rate Limits

| API | Rate Limit | TokenBucket Setup |
|-----|------------|-------------------|
| Exchange orders | 10/sec | `TokenBucket(10, 10)` |
| Market data | 100/min | `rate_limit(100, 60)` |
| Twitter | 300/15min | `rate_limit(300, 900)` |
| OpenAI | 60/min | `rate_limit(60, 60)` |

## Token vs Leaky Bucket

| Feature | Token | Leaky |
|---------|-------|-------|
| Burst support | Yes | No |
| Simple | Yes | Yes |
| Smooth output | No | Yes |
| Memory | Tokens | Queue |

## Quick Reference

```python
# Create
limiter = rate_limit(10, per_seconds=1.0)  # 10/sec
limiter = TokenBucket(rate=10, capacity=20)  # 10/sec, burst 20

# Use
if limiter.try_acquire():        # Non-blocking
    make_request()

with limiter.limit():            # Block until available
    make_request()

limiter.acquire(timeout=1.0)     # With timeout

# Stats
limiter.stats()
limiter.wait_time()              # Time until next token
```
