Calculate probability of losing your account given trading statistics. Essential for capital preservation.

## The Concept

**Risk of Ruin (RoR)** = Probability of losing X% of capital before reaching target gain.

Formula (simplified):
```
RoR = ((1 - edge) / (1 + edge))^(capital_units)
```

Where edge = win% - loss% (adjusted for payoff)

## Implementation

```python
"""
Risk of Ruin Calculator
Probability analysis for capital preservation.
"""

from dataclasses import dataclass
from decimal import Decimal
from typing import Optional
import math


@dataclass
class TradingStats:
    """Trading statistics for RoR calculation."""
    win_rate: Decimal        # 0.60 = 60%
    avg_win: Decimal         # $ per winning trade
    avg_loss: Decimal        # $ per losing trade
    trades_per_period: int = 100  # Trades per period (year, month, etc.)
    
    @property
    def payoff_ratio(self) -> Decimal:
        """Avg win / avg loss."""
        return abs(self.avg_win / self.avg_loss) if self.avg_loss != 0 else Decimal("0")
    
    @property
    def expectancy(self) -> Decimal:
        """Expected $ per trade."""
        return self.win_rate * self.avg_win + (1 - self.win_rate) * (-self.avg_loss)
    
    @property
    def expectancy_ratio(self) -> Decimal:
        """Expectancy as ratio of avg loss."""
        return self.expectancy / abs(self.avg_loss) if self.avg_loss != 0 else Decimal("0")
    
    @property
    def edge(self) -> Decimal:
        """Statistical edge (expectancy ratio)."""
        return self.expectancy_ratio


@dataclass
class RuinResult:
    """Risk of ruin calculation result."""
    ruin_probability: Decimal
    ruin_pct: Decimal          # % loss considered ruin
    capital_units: Decimal     # Account / risk per trade
    edge: Decimal
    win_rate: Decimal
    payoff_ratio: Decimal
    
    # Survival metrics
    half_life_trades: Optional[int]  # Trades until 50% ruin chance
    expected_survival: int           # Trades before hitting ruin threshold
    
    def risk_level(self) -> str:
        """Classify risk level."""
        p = float(self.ruin_probability)
        if p < 0.01:
            return "VERY LOW"
        elif p < 0.05:
            return "LOW"
        elif p < 0.15:
            return "MODERATE"
        elif p < 0.30:
            return "HIGH"
        else:
            return "EXTREME"
    
    def __str__(self) -> str:
        lines = [
            "=== RISK OF RUIN ANALYSIS ===",
            f"Ruin Probability: {self.ruin_probability:.2%}",
            f"Risk Level:       {self.risk_level()}",
            f"",
            f"Ruin Threshold:   {self.ruin_pct:.0%} loss",
            f"Capital Units:    {self.capital_units:.1f}R",
            f"",
            f"Win Rate:         {self.win_rate:.1%}",
            f"Payoff Ratio:     {self.payoff_ratio:.2f}",
            f"Edge:             {self.edge:.2%} per trade",
        ]
        if self.half_life_trades:
            lines.append(f"")
            lines.append(f"50% Ruin Chance:  ~{self.half_life_trades} trades")
        return "\n".join(lines)


class RiskOfRuinCalculator:
    """
    Calculate risk of ruin probability.
    
    Methods:
    - Kelly-based: Uses optimal f derivation
    - Monte Carlo: Simulates many trading sequences
    - Analytical: Closed-form probability
    """
    
    def __init__(self, stats: TradingStats):
        self.stats = stats
    
    def calculate(
        self,
        account_size: Decimal,
        risk_per_trade: Decimal,
        ruin_threshold: Decimal = Decimal("1.0")  # 100% = total loss
    ) -> RuinResult:
        """
        Calculate risk of ruin.
        
        Args:
            account_size: Total capital
            risk_per_trade: $ risked per trade
            ruin_threshold: % loss considered ruin (1.0 = 100%, 0.5 = 50%)
        """
        # Capital units (how many R to ruin)
        capital_units = (account_size * ruin_threshold) / risk_per_trade
        
        # Edge per trade
        edge = self.stats.edge
        
        # Analytical RoR formula
        if edge <= 0:
            # Negative expectancy = eventual ruin
            ruin_prob = Decimal("1.0")
        else:
            # Simplified Kelly-based RoR
            # RoR = ((1-edge)/(1+edge))^capital_units
            numerator = 1 - edge
            denominator = 1 + edge
            
            if denominator > 0 and numerator > 0:
                ratio = numerator / denominator
                ruin_prob = ratio ** capital_units
            else:
                ruin_prob = Decimal("1.0")
        
        # Half-life (trades until 50% ruin probability)
        half_life = None
        if edge > 0 and ruin_prob < Decimal("0.5"):
            # Solve for n where ratio^n = 0.5
            ratio = numerator / denominator
            if ratio > 0:
                half_life = int(math.log(0.5) / math.log(float(ratio)))
        
        # Expected survival (mean trades to ruin)
        expected = self._expected_survival(capital_units, edge)
        
        return RuinResult(
            ruin_probability=ruin_prob,
            ruin_pct=ruin_threshold,
            capital_units=capital_units,
            edge=edge,
            win_rate=self.stats.win_rate,
            payoff_ratio=self.stats.payoff_ratio,
            half_life_trades=half_life,
            expected_survival=expected
        )
    
    def _expected_survival(self, units: Decimal, edge: Decimal) -> int:
        """Expected number of trades before hitting ruin."""
        if edge <= 0:
            return int(units)  # Approximate
        # Random walk expected time to absorption
        return int(units / float(edge))
    
    def monte_carlo(
        self,
        account_size: Decimal,
        risk_per_trade: Decimal,
        ruin_threshold: Decimal = Decimal("1.0"),
        simulations: int = 10000,
        max_trades: int = 1000
    ) -> Decimal:
        """
        Monte Carlo simulation for RoR.
        
        More accurate for complex scenarios.
        """
        import random
        
        ruin_count = 0
        ruin_level = account_size * ruin_threshold
        
        for _ in range(simulations):
            capital = account_size
            for _ in range(max_trades):
                # Simulate trade
                if random.random() < float(self.stats.win_rate):
                    pnl = self.stats.avg_win
                else:
                    pnl = -self.stats.avg_loss
                
                # Adjust to risk per trade
                pnl = pnl * (risk_per_trade / abs(self.stats.avg_loss))
                capital += pnl
                
                # Check ruin
                if (account_size - capital) >= ruin_level:
                    ruin_count += 1
                    break
                
                # Check if could keep trading indefinitely
                if capital > account_size * 2:  # Doubled account
                    break
        
        return Decimal(ruin_count) / Decimal(simulations)
    
    def optimal_risk(
        self,
        account_size: Decimal,
        target_ror: Decimal = Decimal("0.01")  # 1% max RoR
    ) -> dict:
        """
        Find max risk per trade for target RoR.
        
        Returns recommended risk as % of account.
        """
        edge = self.stats.edge
        
        if edge <= 0:
            return {
                "risk_pct": Decimal("0"),
                "reason": "Negative expectancy - any risk leads to ruin"
            }
        
        # Solve for capital_units where RoR = target
        # target = ratio^units
        # units = log(target) / log(ratio)
        numerator = 1 - edge
        denominator = 1 + edge
        ratio = numerator / denominator
        
        if ratio <= 0:
            return {
                "risk_pct": Decimal("0"),
                "reason": "Invalid ratio"
            }
        
        # Capital units needed
        units = Decimal(math.log(float(target_ror)) / math.log(float(ratio)))
        
        # Convert to % of account
        if units > 0:
            risk_pct = Decimal("1") / units  # 1/unit of account per trade
        else:
            risk_pct = Decimal("0")
        
        # Cap at Kelly
        kelly = edge
        if risk_pct > kelly:
            risk_pct = kelly * Decimal("0.5")  # Half Kelly max
            note = "Capped at half Kelly"
        else:
            note = "Within safe limits"
        
        return {
            "risk_pct": risk_pct,
            "risk_dollars": risk_pct * account_size,
            "capital_units": units,
            "kelly": kelly,
            "note": note
        }


# === Quick Functions ===

def risk_of_ruin(
    win_rate: float,
    payoff_ratio: float,
    capital_units: float
) -> float:
    """One-liner RoR calculation."""
    edge = win_rate - (1 - win_rate) / payoff_ratio
    if edge <= 0:
        return 1.0
    ratio = (1 - edge) / (1 + edge)
    return ratio ** capital_units


def max_safe_risk(
    win_rate: float,
    payoff_ratio: float,
    max_ror: float = 0.01
) -> float:
    """Max % of account to risk per trade for target RoR."""
    edge = win_rate - (1 - win_rate) / payoff_ratio
    if edge <= 0:
        return 0.0
    ratio = (1 - edge) / (1 + edge)
    units = math.log(max_ror) / math.log(ratio)
    return 1.0 / units
```
