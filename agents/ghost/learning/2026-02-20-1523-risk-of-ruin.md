# Risk of Ruin Calculator — Account Survival Probability

**Purpose:** Calculate probability of blowing up account given win rate, risk per trade, and edge  
**Use Case:** Position sizing sanity check, strategy validation, risk management

## The Code

```python
"""
Risk of Ruin Calculator
Calculate probability of account destruction given trading parameters.
Essential for position sizing and strategy validation.
"""

from dataclasses import dataclass
from typing import Optional, List, Tuple
from enum import Enum, auto
import math


class RuinModel(Enum):
    """Models for risk of ruin calculation."""
    GAMBLERS_RUIN = auto()  # Fixed bet, binary outcome
    BROWNIAN_MOTION = auto()  # Continuous approximation
    MONTE_CARLO = auto()  # Simulation-based
    VINCE = auto()  # Ralph Vince's formula


@dataclass
class RuinParameters:
    """Input parameters for risk calculation."""
    win_rate: float  # 0.0 to 1.0
    avg_win: float  # Average winning trade ($ or R)
    avg_loss: float  # Average losing trade ($ or R, positive number)
    risk_per_trade_pct: float  # % of account risked per trade
    max_risk: float  # % drawdown considered "ruin" (usually 100%)
    account_size: float = 100000.0  # Starting capital
    
    @property
    def payoff_ratio(self) -> float:
        return self.avg_win / self.avg_loss if self.avg_loss > 0 else 0
    
    @property
    def edge(self) -> float:
        """Mathematical edge (expectation per trade)."""
        return (self.win_rate * self.avg_win) - ((1 - self.win_rate) * self.avg_loss)
    
    @property
    def expectancy(self) -> float:
        """Expectancy as % of risk."""
        return self.payoff_ratio * self.win_rate - (1 - self.win_rate)


@dataclass
class RuinResult:
    """Risk of ruin calculation result."""
    risk_of_ruin_pct: float  # 0-100%
    model_used: RuinModel
    
    # Additional metrics
    break_even_win_rate: float  # Win rate needed to break even
    optimal_risk_pct: float  # Risk level for minimal ruin with growth
    kelly_pct: float  # Kelly criterion optimal bet
    half_kelly_ruin: float  # Ruin probability at half Kelly
    
    # Safety metrics
    safe_risk_level: float  # Risk for <1% ruin probability
    survival_95: int  # Trades until 95% survival probability drops
    
    recommendation: str


class RiskOfRuinCalculator:
    """
    Calculate probability of account destruction.
    
    Usage:
        calc = RiskOfRuinCalculator()
        params = RuinParameters(
            win_rate=0.55,
            avg_win=200,
            avg_loss=100,
            risk_per_trade_pct=2.0
        )
        result = calc.calculate(params)
        print(f"Risk of ruin: {result.risk_of_ruin_pct:.2f}%")
    """
    
    def calculate(
        self,
        params: RuinParameters,
        model: RuinModel = RuinModel.GAMBLERS_RUIN
    ) -> RuinResult:
        """Calculate risk of ruin using specified model."""
        if model == RuinModel.GAMBLERS_RUIN:
            ruin_pct = self._gamblers_ruin(params)
        elif model == RuinModel.BROWNIAN_MOTION:
            ruin_pct = self._brownian_motion(params)
        elif model == RuinModel.VINCE:
            ruin_pct = self._vince_formula(params)
        else:
            ruin_pct = self._gamblers_ruin(params)
        
        # Calculate supporting metrics
        break_even = 1 / (1 + params.payoff_ratio) if params.payoff_ratio > 0 else 0.5
        kelly = params.win_rate - ((1 - params.win_rate) / params.payoff_ratio) if params.payoff_ratio > 0 else 0
        half_kelly_ruin = self._gamblers_ruin(RuinParameters(
            win_rate=params.win_rate,
            avg_win=params.avg_win,
            avg_loss=params.avg_loss,
            risk_per_trade_pct=kelly * 50,  # Half Kelly
            max_risk=params.max_risk,
            account_size=params.account_size
        ))
        
        optimal_risk = self._find_optimal_risk(params)
        safe_risk = self._find_safe_risk(params)
        survival_95 = self._calculate_survival_horizon(params)
        
        recommendation = self._generate_recommendation(ruin_pct, params, kelly)
        
        return RuinResult(
            risk_of_ruin_pct=ruin_pct,
            model_used=model,
            break_even_win_rate=break_even,
            optimal_risk_pct=optimal_risk,
            kelly_pct=kelly * 100,
            half_kelly_ruin=half_kelly_ruin,
            safe_risk_level=safe_risk,
            survival_95=survival_95,
            recommendation=recommendation
        )
    
    def _gamblers_ruin(self, params: RuinParameters) -> float:
        """
        Gambler's Ruin formula for fixed bet size.
        
        R = [(1 - W) / W] ^ (C / RPT)
        Where:
            W = win rate
            C = max risk (ruin threshold)
            RPT = risk per trade
        """
        if params.win_rate <= 0 or params.win_rate >= 1:
            return 100.0 if params.edge < 0 else 0.0
        
        if params.edge <= 0:
            return 100.0  # Negative edge = certain ruin eventually
        
        # Adjust for payoff ratio
        p = params.win_rate
        q = 1 - p
        b = params.payoff_ratio
        
        # Formula with payoff ratio consideration
        if b == 1.0:
            # Simple case: equal wins/losses
            if q >= p:
                return 100.0
            ratio = q / p
            exponent = params.max_risk / params.risk_per_trade_pct
            ruin = ratio ** exponent
        else:
            # Complex case: different win/loss sizes
            # Approximation using effective win probability
            p_eff = (b * p) / (b * p + q)
            q_eff = 1 - p_eff
            
            if q_eff >= p_eff:
                return 100.0
            
            ratio = q_eff / p_eff
            exponent = params.max_risk / params.risk_per_trade_pct
            ruin = ratio ** exponent
        
        return min(100.0, max(0.0, ruin * 100))
    
    def _brownian_motion(self, params: RuinParameters) -> float:
        """
        Brownian motion approximation for continuous trading.
        More accurate for high-frequency trading.
        """
        mu = params.edge / params.avg_loss  # Drift (normalized)
        sigma = math.sqrt(
            params.win_rate * ((params.avg_win/params.avg_loss) - mu) ** 2 +
            (1 - params.win_rate) * (1 - mu) ** 2
        ) if params.avg_loss > 0 else 0.1
        
        if sigma == 0:
            return 100.0 if mu <= 0 else 0.0
        
        # Ruin probability formula
        if mu <= 0:
            return 100.0
        
        # Distance to ruin in standard deviations
        distance = (params.max_risk / params.risk_per_trade_pct) * (mu / sigma)
        
        # Approximate ruin probability
        ruin = math.exp(-2 * distance)
        
        return min(100.0, ruin * 100)
    
    def _vince_formula(self, params: RuinParameters) -> float:
        """
        Ralph Vince's risk of ruin formula.
        Accounts for variable bet sizing.
        """
        # Simplified Vince approximation
        z = params.payoff_ratio
        a = params.win_rate
        
        if z <= 0 or a <= 0:
            return 100.0
        
        # Risk adjusted for variance
        variance = a * (1 - a) * (z + 1) ** 2
        risk_adjusted = params.risk_per_trade_pct * math.sqrt(variance)
        
        # Ruin formula
        if params.edge <= 0:
            return 100.0
        
        safety_factor = params.max_risk / risk_adjusted if risk_adjusted > 0 else 1
        ruin = math.exp(-safety_factor * params.edge / params.avg_loss)
        
        return min(100.0, max(0.0, (1 - ruin) * 100))
    
    def _find_optimal_risk(self, params: RuinParameters) -> float:
        """Find risk level that minimizes ruin while allowing growth."""
        # Grid search for best risk level
        best_risk = 0.5
        best_score = -float('inf')
        
        for risk in [0.5, 1.0, 1.5, 2.0, 2.5, 3.0]:
            test_params = RuinParameters(
                win_rate=params.win_rate,
                avg_win=params.avg_win,
                avg_loss=params.avg_loss,
                risk_per_trade_pct=risk,
                max_risk=params.max_risk,
                account_size=params.account_size
            )
            
            ruin = self._gamblers_ruin(test_params)
            growth = self._estimate_growth_rate(test_params)
            
            # Score: high growth, low ruin
            score = growth * (100 - ruin)
            
            if score > best_score and ruin < 10:  # Must have <10% ruin
                best_score = score
                best_risk = risk
        
        return best_risk
    
    def _find_safe_risk(self, params: RuinParameters) -> float:
        """Find maximum risk for <1% ruin probability."""
        for risk in [5.0, 4.0, 3.0, 2.5, 2.0, 1.5, 1.0, 0.5]:
            test_params = RuinParameters(
                win_rate=params.win_rate,
                avg_win=params.avg_win,
                avg_loss=params.avg_loss,
                risk_per_trade_pct=risk,
                max_risk=params.max_risk,
                account_size=params.account_size
            )
            
            ruin = self._gamblers_ruin(test_params)
            if ruin < 1.0:
                return risk
        
        return 0.5
    
    def _estimate_growth_rate(self, params: RuinParameters) -> float:
        """Estimate expected growth rate per trade."""
        if params.avg_loss <= 0:
            return 0.0
        
        # Kelly-based growth estimate
        kelly = params.win_rate - ((1 - params.win_rate) / params.payoff_ratio) if params.payoff_ratio > 0 else 0
        risk_fraction = (params.risk_per_trade_pct / 100) / (kelly if kelly > 0 else 0.02)
        
        # Expected geometric growth
        growth = params.edge * risk_fraction / params.account_size
        return growth
    
    def _calculate_survival_horizon(self, params: RuinParameters) -> int:
        """Calculate trades until 95% survival probability."""
        if params.edge <= 0:
            return 0
        
        # Approximate using exponential decay
        ruin_prob = self._gamblers_ruin(params) / 100
        if ruin_prob >= 0.95:
            return 0
        
        # Trades until 95% cumulative survival
        # (1 - ruin_prob) ^ n = 0.95
        # n = log(0.95) / log(1 - ruin_prob)
        if ruin_prob > 0:
            n = math.log(0.95) / math.log(1 - ruin_prob)
            return int(n)
        
        return 10000  # Essentially infinite
    
    def _generate_recommendation(self, ruin_pct: float, params: RuinParameters, kelly: float) -> str:
        """Generate human-readable recommendation."""
        if ruin_pct >= 50:
            return f"🛑 CRITICAL: {ruin_pct:.1f}% ruin risk. Reduce risk immediately or stop trading."
        
        if ruin_pct >= 20:
            return f"⚠️  HIGH RISK: {ruin_pct:.1f}% ruin probability. Reduce to {params.risk_per_trade_pct * 0.5:.1f}% risk per trade."
        
        if ruin_pct >= 5:
            return f"⚡ ELEVATED: {ruin_pct:.1f}% ruin risk. Consider reducing to {kelly * 25:.1f}% (quarter Kelly)."
        
        if ruin_pct >= 1:
            return f"✅ ACCEPTABLE: {ruin_pct:.2f}% ruin risk. Monitor for strategy degradation."
        
        return f"🎯 EXCELLENT: {ruin_pct:.3f}% ruin risk. Well within safe parameters."
    
    def simulate_ruin(
        self,
        params: RuinParameters,
        num_simulations: int = 1000,
        max_trades: int = 1000
    ) -> Tuple[float, List[int]]:
        """
        Monte Carlo simulation of ruin probability.
        Returns (ruin_pct, list of trades_to_ruin for ruined accounts).
        """
        import random
        
        ruined = 0
        trades_to_ruin = []
        
        for _ in range(num_simulations):
            account = params.account_size
            ruin_threshold = params.account_size * (1 - params.max_risk / 100)
            
            for trade_num in range(1, max_trades + 1):
                # Simulate trade
                if random.random() < params.win_rate:
                    # Win
                    pnl = params.avg_win * (account * params.risk_per_trade_pct / 100) / params.avg_loss
                else:
                    # Loss
                    pnl = -(account * params.risk_per_trade_pct / 100)
                
                account += pnl
                
                if account <= ruin_threshold:
                    ruined += 1
                    trades_to_ruin.append(trade_num)
                    break
        
        ruin_pct = (ruined / num_simulations) * 100
        return ruin_pct, trades_to_ruin


def quick_ruin_check(win_rate: float, risk_reward: float, risk_pct: float) -> str:
    """
    Quick risk of ruin estimate.
    
    Args:
        win_rate: 0.0 to 1.0
        risk_reward: Average win / Average loss
        risk_pct: % of account risked per trade
    
    Returns: Risk assessment string
    """
    params = RuinParameters(
        win_rate=win_rate,
        avg_win=risk_reward,
        avg_loss=1.0,
        risk_per_trade_pct=risk_pct,
        max_risk=100.0
    )
    
    calc = RiskOfRuinCalculator()
    result = calc.calculate(params)
    
    return f"Risk of Ruin: {result.risk_of_ruin_pct:.2f}% | {result.recommendation}"


def optimal_position_sizing(win_rate: float, avg_win: float, avg_loss: float) -> dict:
    """Recommend position sizing based on ruin minimization."""
    calc = RiskOfRuinCalculator()
    
    results = {}
    for risk in [0.5, 1.0, 2.0, 3.0, 5.0]:
        params = RuinParameters(
            win_rate=win_rate,
            avg_win=avg_win,
            avg_loss=avg_loss,
            risk_per_trade_pct=risk,
            max_risk=100.0
        )
        result = calc.calculate(params)
        results[risk] = result.risk_of_ruin_pct
    
    # Find best risk level
    safe_options = {k: v for k, v in results.items() if v < 5.0}
    if safe_options:
        best_risk = max(safe_options.keys())  # Highest safe risk
    else:
        best_risk = min(results.keys())  # Lowest risk
    
    return {
        "recommended_risk_pct": best_risk,
        "recommended_ruin_prob": results[best_risk],
        "all_options": results,
        "kelly_pct": result.kelly_pct if 'result' in locals() else 0
    }

def find_min_win_rate(risk_pct: float, payoff_ratio: float, max_ruin: float, calc: RiskOfRuinCalculator) -> float:
    """Find minimum win rate for given risk and payoff."""
    for wr in [x / 100 for x in range(30, 71)]:
        params = RuinParameters(
            win_rate=wr,
            avg_win=payoff_ratio,
            avg_loss=1.0,
            risk_per_trade_pct=risk_pct,
            max_risk=100.0
        )
        result = calc.calculate(params)
        if result.risk_of_ruin_pct <= max_ruin:
            return wr
    return 0.70
```
